304  (defn msg->key
305    "Extracts the `key` from a published keyspace
306    notification message (`pmessage`).
307  
308    Example:
309    ```clojure
310    (def msg [\"pmessage\"
311             \"__keyspace@0*__:ref@*@*@ctrl*\"
312             \"__keyspace@0__:ref@container@0@ctrl\"
313             \"set\"])
314    (st/msg->key msg)
315    ;; \"ref@container@0@ctrl\"
316    ```"
317    [[kind l1 l2 l3]]
318    (timbre/debug "msg->key received: " (str kind l1 l2 l3))
319    (condp = (keyword kind)
320      :pmessage   (second (string/split l2 #":"))
321      :psubscribe (timbre/info "subscribed to " l1)
322      (timbre/warn "received" kind l1 l2 l3)))

324  (defn subs-pat
325    "Generates subscribe patterns which matches
326    depending on:
327    
328    **l2**
329    
330    * `container`
331    * `definitions`
332    
333    **l3**
334    
335    * `0` ... `n`
336  
337    **l4**
338    
339    * `ctrl`
340    * `state`
341    * `definition`
342    "
343    [mp-id l2 l3 l4]
344    (str "__keyspace@0*__:" mp-id
345         u/sep l2
346         u/sep l3
347         u/sep l4 "*"))

349  (defn gen-listener
350    "Returns a listener for published keyspace
351    notifications. Don't forget to [[close-listener!]]
352  
353    Example:
354    ```clojure
355    ;; generate and close
356    (close-listener! (gen-listener \"ref\" \"ctrl\" msg->key))
357    ```"
358    [mp-id l2 l3 l4 cb!]
359    (let [s-pat (subs-pat mp-id l2 l3 l4)]
360      (timbre/debug "new pubsub-listener for pat: " s-pat)
361      (car/with-new-pubsub-listener (:spec conn)
362        {s-pat cb!}
363        (car/psubscribe s-pat))))  
364  
365  (defn close-listener!
366    "Closes the given listener generated by [[gen-listener]].
367  
368    Example:
369    ```clojure
370    ;; generate
371    (def l (gen-listener \"ref\" \"ctrl\" msg->key))
372    ;; close 
373    (close-listener! l)
374    ```"
375    [l]
376    (car/close-listener l))

386  (defn reg-key
387    "Generates a registration key for the listener atom."
388    [mp-id struct no func]
389    (str mp-id "_" struct "_" no "_" func))
390  
391  (defn registered?
392    "Checks if a `listener` is registered under
393    the `listeners`-atom."
394    [reg-key]
395    (contains? (deref listeners) reg-key))
396  
397  (defn register!
398    "Generates and registers a  listener under the
399    key `mp-id` in the `listeners` atom.
400    The cb! function dispatches depending on
401    the result."
402    [mp-id struct no func cb!]
403    (let [reg-key (reg-key mp-id struct no func)]
404      (if-not (registered? reg-key)
405        {:ok (map?
406              (swap! listeners assoc
407                     reg-key
408                     (gen-listener mp-id struct no func cb!)))}
409        {:ok true :warn "already registered"})))
410    
411  (defn de-register!
412    "De-registers the listener with the
413    key `mp-id` in the `listeners` atom."
414    [mp-id struct no func]
415    (let [reg-key (reg-key mp-id struct no func)]
416      (if (registered? reg-key)
417        (do
418          (timbre/debug "de-register:" reg-key)
419          (close-listener! ((deref listeners) reg-key))
420          {:ok (map?
421                (swap! listeners dissoc
422                       reg-key))})
423        {:ok true :warn "not registered"})))
