016  (defn set-val!
017    "Sets the value `v` for the key `k`."
018    [k v]
019    (if (string? k)
020      (if (some? v)
021        (wcar conn (car/set k (che/generate-string v)))
022        (log/warn "no value given"))
023      (log/warn "no key given")))
024  
025  (defn set-same-val!
026    "Sets the given `val` for all keys `ks` with the delay `mtp`."
027    [ks v]
028    (run! (fn [k] (set-val! k v)) ks))

033  (defn set-state!
034    "Function is used by the workers to set state. An optional log message
035    may be provided."
036    ([k state msg]
037     (condp = state
038       :error (log/error msg)
039       :ready (log/debug msg)
040       (log/debug msg))
041     (set-state! k state))
042    ([k state]
043     (when (and (string? k)
044                (keyword? state)) 
045       (set-val! k (name state))
046       (log/debug "wrote new state: " state " to: " k)
047       (when (= state :working)
048         (log/debug "just set working state")
049         (Thread/sleep mtp)
050         (log/debug "relaxation time over")))))

097  (defn filter-keys-where-val
098    "Returns a list of all keys belonging to the pattern `pat` where the
099    value is equal to`x`.
100    
101    Example:
102    ```clojure
103    (filter-keys-where-val \"ref@definitions@*@class\" \"wait\")
104    ;; (\"ref@definitions@0@class\"
105    ;; \"ref@definitions@2@class\"
106    ;; \"ref@definitions@1@class\")
107    ```
108    "

110    (filter (fn [k] (= x (key->val k))) (pat->keys pat)))

115  (defn msg->key
116    "Extracts the `key` from a published keyspace notification
117    message (`pmessage`).
118  
119    Example:
120    ```clojure
121    (def msg [\"pmessage\"
122             \"__keyspace@0*__:ref@*@*@ctrl*\"
123             \"__keyspace@0__:ref@container@0@ctrl\"
124             \"set\"])
125    (st/msg->key msg)
126    ;; \"ref@container@0@ctrl\"
127    ```"
128    [[kind l1 l2 l3]]
129    (condp = (keyword kind)
130      :pmessage   (second (string/split l2 #":"))
131      :psubscribe (log/debug "subscribed to " l1)
132      (log/warn "received" kind l1 l2 l3)))

134  (defn subs-pat
135    "Generates subscribe patterns which matches depending on:
136    
137    **l2**
138    
139    * `container`
140    * `definitions`
141    
142    **l3**
143    
144    * `0` ... `n`
145  
146    **l4**
147    
148    * `ctrl`
149    * `state`
150    * `definition`
151    "
152    [mp-id l2 l3 l4]
153    (str "__keyspace@" db "*__:" mp-id
154         u/sep l2
155         u/sep l3
156         u/sep l4 "*"))

158  (defn gen-listener
159    "Returns a listener for published keyspace notifications. Don't forget
160    to [[close-listener!]]
161  
162    Example:
163    ```clojure
164    ;; generate and close
165    (close-listener! (gen-listener \"ref\" \"ctrl\" msg->key))
166    ```"
167    [mp-id l2 l3 l4 callback]
168    (let [pat (subs-pat mp-id l2 l3 l4)]
169      (car/with-new-pubsub-listener (:spec conn)
170        {pat callback}
171        (car/psubscribe pat))))  
172  
173  (defn close-listener!
174    "Closes the given listener generated by [[gen-listener]].
175  
176    Example:
177    ```clojure
178    ;; generate
179    (def l (gen-listener \"ref\" \"ctrl\" msg->key))
180    ;; close 
181    (close-listener! l)
182    ```"
183    [l]
184    (car/close-listener l))

194  (defn reg-key
195    "Generates a registration key for the listener atom.
196    The `level` param allows to register more than one listener for one
197    pattern."
198    [mp-id struct no func level]
199    (str mp-id "_" struct "_" no "_" func "_" level))
200  
201  (defn registered?
202    "Checks if a `listener` is registered under
203    the `listeners`-atom."
204    [k]
205    (contains? (deref listeners) k))
206  
207  (defn register!
208    "Generates and registers a listener under the key `mp-id` in the
209    `listeners` atom.  The cb! function dispatches depending on the
210    result."
211    ([mp-id struct no func cb!]
212     (register! mp-id struct no func cb! "a"))
213    ([mp-id struct no func cb! level]
214     (let [reg-key (reg-key mp-id struct no func level)]
215       (if-not (registered? reg-key)
216         {:ok (map?
217               (swap! listeners assoc
218                      reg-key
219                      (gen-listener mp-id struct no func cb!)))}
220         {:ok true :warn "already registered"}))))
221  
222  (defn de-register!
223    "De-registers the listener with the key `mp-id` in the `listeners`
224    atom."
225    ([mp-id struct no func]
226     (de-register! mp-id struct no func "a"))
227    ([mp-id struct no func level]
228     (let [k (reg-key mp-id struct no func level)]
229       (if (registered? k)
230         (do
231           (log/debug "de-register:" reg-key)
232           (close-listener! ((deref listeners) k))
233           {:ok (map? (swap! listeners dissoc k))})
234         {:ok true :warn "not registered"}))))
235  
236  (defn clean-register!
237    "Closes and `de-registers!` all `listeners` belonging to `mp-id` ."
238    [mp-id]
239    (map (fn [[k v]]
240           (if (string/starts-with? k mp-id)
241             (do
242               (close-listener! v)
243               {:ok (map? (swap! listeners dissoc k))})
244             {:ok true :reason "unrelated"}))
245         (deref listeners)))
