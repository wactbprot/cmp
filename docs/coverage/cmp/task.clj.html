029  (defn task?
030    "Checks the task structure against a spec."
031    [m]
032    (if (nil? m)
033      (throw (Exception. "task is nil"))
034      (condp :Action m
035        "TCP"    (s/valid? ::tcp-task m)
036        :default (s/valid? ::task m))))

096  (defn outer-replace-map
097    "Replaces tokens (given in the m) in the task.
098    This kind of replacement is used during the
099    task build up at the beginning of its life cycle.
100    
101    Example:
102    ```clojure
103    (outer-replace-map (globals) {:TaskName \"foo\" :Value \"%time\"})
104    ;; {:TaskName \"foo\", :Value \"1580652820247\"}
105    (outer-replace-map nil {:TaskName \"foo\" :Value \"%time\"})
106    ;; {:TaskName \"foo\", :Value \"%time\"}
107    ```
108    "
109    [m task]
110    (if (map? m)
111      (u/json->map
112       (reduce
113        (fn [s [k v]]
114          (let [pat (re-pattern (name k))
115                r   (u/clj->str-val v)]
116            (string/replace s pat r)))
117        (u/map->json task) m))
118      task))

120  (defn inner-replace-map
121    "Applies the generated function  `f` to the
122    values `v` of the of the `task` map. `f`s input is `v`.
123    If `m` has a key `v` the value of this key is returned.
124    If `m` has no key `v` the `v` returned.
125    This kind of replacement is used during the
126    runtime.
127    "
128    [m task]
129    (let [nm (u/apply-to-map-keys name m)
130          f (fn [v]
131              (if-let [r (get nm  v)]
132                r
133                v))]
134      (u/apply-to-map-values f task)))

141  (defn str->singular-kw
142    "Takes a keyword or string and removes the tailing
143    letter (most likely a s). Turns the result
144    to a keyword.
145    
146    ```clojure
147    (str->singular-kw :Values)
148    ;; :Value
149    (str->singular-kw \"Values\")
150    ;; :Value
151    ```
152    "
153    [s]
154    (->> s
155         name
156         (re-matches #"^(\w*)(s)$")
157         second
158         keyword))

160  (defn merge-use-map
161    "The use keyword enables a replace mechanism.
162    It works like this:
163    proto-task:
164    
165    ```clojure
166    Use: {Values: med_range}
167    ;; should lead to:
168    task: { Value: rangeX.1}
169    ```"
170    [m task]
171    (if (map? m)
172      (merge task
173             (into {}
174                   (map
175                    (fn [k]
176                      (hash-map
177                       (str->singular-kw k)
178                       (extract-use-value task m k)))
179                    (keys m))))
180      task))

189  (defn gen-meta-task
190    "Gathers all information for the given `proto-task` (map).
191    The `proto-task` should be a map containing the `:TaskName`
192    `keyword` at least. String version makes a `map` out of `s`
193    and calls related method.
194  
195    ```clojure
196    (gen-meta-task \"Common-wait\")
197    ;; 19-12-27 11:14:48 hiob DEBUG [cmp.lt-mem:21] - get task:  Common-wait  from ltm
198    ;; {:Task
199    ;; {:Action \"wait\",
200    ;; :Comment \"%waitfor  %waittime ms\",
201    ;; :TaskName \"Common-wait\",
202    ;; :WaitTime \"%waittime\"},
203    ;; :Use nil,
204    ;; :Defaults
205    ;; {\"%unit\" \"mbar\",
206    ;; \"%targetdb\" \"vl_db\",
207    ;; \"%relayinfo\" \"relay_info\",
208    ;; \"%docpath\" \"\",
209    ;; \"%sourcedb\" \"vl_db_work\",
210    ;; \"%timepath\" \"Time\",
211    ;; \"%waitunit\" \"ms\",
212    ;; \"%break\" \"no\",
213    ;; \"%waitfor\" \"Ready in\",
214    ;; \"%waittime\" \"1000\",
215    ;; \"%dbinfo\" \"db_info\"},
216    ;; :Globals
217    ;; {\"%hour\" \"11\",
218    ;; \"%minute\" \"14\",
219    ;; \"%second\" \"48\",
220    ;; \"%year\" \"2019\",
221    ;; \"%month\" \"12\",
222    ;; \"%day\" \"27\",
223    ;; \"%time\" \"1577445288247\"},
224    ;; :Replace nil}
225    ;;
226    ;; call the map vesion as follows:
227    
228    (gen-meta-task {:TaskName \"Common-wait\" :Replace {\"%waittime\" 10}})
229    ```"
230    [x]
231    (let  [proto-task (ensure-proto-task x)
232           task-name  (:TaskName proto-task)
233           db-task    (merge
234                       (->> ["tasks" task-name]
235                            u/vec->key
236                            st/key->val)
237                       proto-task)]
238      {:Task          (dissoc db-task :Defaults) 
239       :Use           (:Use proto-task)
240       :Globals       (globals)
241       :Defaults      (:Defaults db-task)
242       :Replace       (:Replace proto-task)}))

244  (defn assemble
245    "Assembles the `task` from the given
246    `meta-task` in a special order:
247  
248    * merge Use
249    * replace from Replace
250    * replace from Defaults
251    * replace from Globals
252    
253    `assoc`s the structs afterwards.
254  
255    ```clojure
256    (def proto-task {:TaskName \"Common-wait\"
257                      :Replace {\"%waittime\" 10}})
258    (assemble
259      (gen-meta-task proto-task \"ref\" \"ref@container@0@state@0@0\"))
260    ;; {:Action  \"wait\",
261    ;;  :Comment  \"Ready in  10 ms\",
262    ;;  :TaskName \"Common-wait\",
263    ;;  :WaitTime \"10\",
264    ;;  :MpName   \"ref\"
265    ;;  :StateKey \"ref@container@0@state@0@0\"
266    ;;  ...
267    ;; }
268    ```
269    "
270    [meta-task mp-id state-key]
271    (let [db-task  (:Task         meta-task)
272          use-map  (:Use          meta-task)
273          replace  (:Replace      meta-task)
274          defaults (:Defaults     meta-task)
275          globals  (:Globals      meta-task)
276          exch-map (:FromExchange db-task)
277          task     (dissoc db-task
278                           :FromExchange
279                           :Replace)
280          from-map (exch/from! mp-id exch-map)]
281      (assoc 
282       (->> task
283            (merge-use-map use-map)
284            (inner-replace-map from-map)
285            (outer-replace-map replace)
286            (outer-replace-map defaults)
287            (outer-replace-map globals))
288       :MpName    mp-id
289       :StateKey  state-key)))
