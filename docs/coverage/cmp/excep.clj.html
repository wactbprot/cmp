001  (ns cmp.excep
002    ^{:author "wactbprot"
003      :doc "Provides an exception channel."}
004    (:require [taoensso.timbre :as timbre]
005              [clojure.core.async :as a]))
006  
007  ;;------------------------------
008  ;; exception channel 
009  ;;------------------------------
010  (def ch (a/chan))
011  (a/go-loop []
012    (let [e (a/<! ch)]
013      (if (string? e)
014        (timbre/error e)
015        (timbre/error (.getMessage e))))
016    (recur))
