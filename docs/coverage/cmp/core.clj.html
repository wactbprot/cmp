181  (defn check
182    "
183    REVIEW
184    
185    Check the tasks of the `container` and
186    `definitions` structure. *cmp* does not preload
187    the tasks, they are loaded from the `lt-mem`
188    during runtime."
189    ([]
190     (check (->mp-id)))
191    ([mp-id]
192     (let [p         (u/main-path mp-id)
193           k-ncont   (st/meta-ncont-path p)
194           n-cont    (st/key->val k-ncont)
195           k-ndefins (st/meta-ndefins-path p)
196           n-defins  (st/key->val k-ndefins)]
197       (run!
198        (fn [i]
199          (chk/struct-tasks (st/cont-defin-path p i)))
200        (range n-cont))
201       (run!
202        (fn [i]
203          (chk/struct-tasks (st/defins-defin-path p i)))
204        (range n-defins)))))

232    "Push a command string (`cmd`) to the control
233    interface of a mp. `cmd`s are:

291  (defn t-table
292    "Prints a table of **assembled tasks** stored in
293    **short term memory**. If a `kw` and `val` are given
294    it is used as a filter
295  
296    Example:
297    ```clojure
298    (t-table)
299    ;; same as
300    (t-table :Action :all)
301    ;;
302    (t-table :Action \"TCP\")
303    ;;
304    (t-table :Port \"23\")
305    ```
306    . "
307    ([]
308     (t-table  :Action :all))
309    ([kw v]
310     (pp/print-table
311      (filter some?
312              (into []
313                    (map (fn [k]
314                           (let [name  (st/key->struct k)
315                                 task  (tsk/assemble (tsk/gen-meta-task name))
316                                 value (kw task)]
317                             (if (and value (or (= value v) (= :all v)))
318                               {:stm-key k :Name name kw value} )))
319                         (st/key->keys "tasks")))))))
320  (defn t-run
321    [name]
322    (-> name
323        tsk/gen-meta-task
324        tsk/assemble
325        (work/dispatch! "core")))
