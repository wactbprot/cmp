042    "Sets the mpd to work on (see [[current-mp-id]]).
043  
044    Usage:
045    
046    ```clojure
047    (workon! 'se3-calib')
048    (->mp-id)
049    ```
050    "
051    [mp-id]
052    (reset! current-mp-id mp-id))

056

058    
059     ```clojure
060    (workon! 'se3-calib')

062    ```
063    "
064    []
065    (if-let [mp-id (deref current-mp-id)]
066      mp-id
067      (throw (Exception. "No mp-id set.\n\n\nUse function (workon! <mp-id>)"))))

112  (defn build-mpd
113    "Loads mpd from long term memory and
114    builds the short term memory. The `mp-id`
115    must be set with [[workon!]].

117    Usage:
118    
119    ```clojure

128    ([]
129     (build-mpd (->mp-id)))

132     (->> mp-id

135          (u/doc->safe-doc)
136          (bld/store))))

173  ;; check mp tasks
174  ;;------------------------------
175  (defn check
176    "Check the tasks of the `container` and
177    `definitions` structure. *cmp* does not preload
178    the tasks, they are loaded from the `lt-mem`
179    during runtime."
180    ([]
181     (check (->mp-id)))
182    ([mp-id]
183     (let [p         (u/extr-main-path mp-id)
184           k-ncont   (st/get-meta-ncont-path p)
185           n-cont    (st/key->val k-ncont)
186           k-ndefins (st/get-meta-ndefins-path p)
187           n-defins  (st/key->val k-ndefins)]
188       (run!
189        (fn [i]
190          (chk/struct-tasks (st/get-cont-defin-path p i)))
191        (range n-cont))
192       (run!
193        (fn [i]
194          (chk/struct-tasks (st/get-defins-defin-path p i)))
195        (range n-defins)))))

223  ;;------------------------------
224  (defn set-ctrl
225    "Push a command string (`cmd`) to the control
226    interface of a mp. `cmd`s are:
227    
228    * `\"run\"`
229    * `\"stop\"`
230    * `\"mon\"`
231    * `\"suspend\"`
232  
233    The `mp-id` is received over `(->mp-id)`.
234    **NOTE:** The `definitions` struct should not
235    be started by user (see [[workon!]])."
236    ([i cmd]
237     (set-ctrl (->mp-id) i cmd))
238    ([mp-id i cmd]
239     (let [p (st/get-cont-ctrl-path (u/extr-main-path mp-id) i)]
240       (st/set-val! p cmd))))
