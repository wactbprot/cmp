227  (defn set-ctrl
228    "Writes the command string (`cmd`) to the control
229    interface of a `mpd`. If the `mpd` is already
230    started (see [[m-start]]) the next steps work
231    as follows:  `cmd` is written to the short
232    term memory by means of [[cmp.st-mem.set-val!]].
233    The writing process triggers the `registered`
234    `callback` (registered by [[m-start]]). The
235    `callback` cares about the `cmd`.  `cmd`s are:
236    
237    * `\"run\"`
238    * `\"stop\"`
239    * `\"mon\"`
240    * `\"suspend\"`
241  
242    The `mp-id` parameter may be give directly
243  
244    ```clojure
245    (set-ctrl \"ref\" \"run\")
246    ```
247    or is derived from `(deref current-mp-id)` [[workon \"ref\"]].
248  
249  
250    ```clojure
251    (workon! \"ref\"=
252    (set-ctrl \"run\")
253    ```
254  
255    **NOTE:**
256  
257    `set-ctrl` only writes to the `container` structure.
258    The `definitions` struct should not be started by a
259    user (see [[workon!]])."
260    ([i cmd]
261     (set-ctrl (deref current-mp-id) i cmd))
262    ([mp-id i cmd]
263     (st/set-val! (st/cont-ctrl-path  mp-id (u/lp i)) cmd)))

349  (defn t-run
350    "Runs the task with the given name (from stm).
351    If only the name is provided, results are stored
352    under  `core@test@0@response@0@0`.
353  
354    If  `mp-id`, `struct`, `i`, `j` and  `k` is given,
355    the results are written to `<mp-id@<struct>@<i>@response@<j>@<k>`.
356    A listener at this key triggers a `cb!` which de-registers
357    and closes the listener. The callback also gets the value of 
358    the key (`<mp-id@<struct>@<i>@response@<j>@<k>`) and pretty
359    prints it.
360  
361    REVIEW (function is way to large)
362    
363    Example:
364    ```clojure
365    (t-run \"DKM_PPC4_DMM-read_temp\")
366    ;;
367    ;; {:t_start 1588071759882,
368    ;; :t_stop 1588071768996,
369    ;; :Result
370    ;; [{:Type dkmppc4,
371    ;;  :Value 24.297828639,
372    ;;  :Unit C,
373    ;;  :SdValue 0.0013625169107,
374    ;;  :N 10}]}
375    ```
376  
377    Debug:
378    ```clojure
379    @st/listeners
380    (st/de-register! \"core\" \"test\" 0 \"response\")
381    ```
382    "
383    ([name]
384     (t-run name "core" "test" 0 0 0))
385    ([name mp-id]
386     (t-run name mp-id "test" 0 0 0))
387    ([name mp-id struct no-idx seq-idx par-idx]
388     (let [no-idx    (u/lp no-idx)
389           seq-idx   (u/lp seq-idx)
390           par-idx   (u/lp par-idx)
391           func       "response"
392           state-key  (u/vec->key [mp-id struct no-idx "state" seq-idx par-idx])
393           resp-key   (u/vec->key [mp-id struct no-idx func    seq-idx par-idx])
394           meta-task  (task/gen-meta-task name)
395           task       (task/assemble meta-task mp-id state-key)]
396       (when (task/dev-action? task)
397         (timbre/info "task dispached, wait for response...")
398         (st/register! mp-id struct no-idx func (fn [msg]
399                                                  (when-let [result-key (st/msg->key msg)]
400                                                    (st/de-register! mp-id struct no-idx func)
401                                                    (pp/pprint (st/key->val result-key))))))
402       (work/check task))))
