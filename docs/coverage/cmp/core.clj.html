230  (defn set-ctrl
231    "Writes the command string (`cmd`) to the control interface of a
232    `mpd`. If the `mpd` is already started (see [[m-start]]) the next
233    steps work as follows: `cmd` is written to the short term memory by
234    means of [[cmp.st-mem.set-val!]].  The writing process triggers the
235    `registered` `callback` (registered by [[m-start]]). The `callback`
236    cares about the `cmd`.  `cmd`s are:
237    
238    * `\"run\"`
239    * `\"stop\"`
240    * `\"mon\"`
241    * `\"suspend\"`
242  
243    The `mp-id` parameter may be give directly
244  
245    ```clojure
246    (set-ctrl \"ref\" \"run\")
247    ```
248    or is derived from `(deref current-mp)` [[workon \"ref\"]].
249  
250  
251    ```clojure
252    (workon! \"ref\"=
253    (set-ctrl \"run\")
254    ```
255  
256    **NOTE:**
257  
258    `set-ctrl` only writes to the `container` structure.  The
259    `definitions` struct should not be started by a
260    user (see [[workon!]])."
261    ([i cmd]
262     (set-ctrl (deref current-mp) i cmd))
263    ([mp-id i cmd]
264     (st/set-val! (ku/cont-ctrl-key  mp-id (u/lp i)) cmd)))

311    "Returns a vector of **assembled tasks** stored in **short term

313  
314    Example:
315    ```clojure
316    (t-data)
317    ;; same as
318    (t-data :Action :all)
319    ;;
320    (t-data :Action \"TCP\")
321    ;;
322    (t-data :Port \"23\" \"ref\")
323    ```
324    . "
325    ([]
326     (t-data  :Action :all "core" "test" 0 0 0))
327    ([kw v]
328     (t-data  kw v "core" "test" 0 0 0))
329    ([kw v mp-id]
330     (t-data  kw v mp-id "test" 0 0 0))
331    ([kw v mp-id struct i j k]
332     (let [state-key (u/vec->key [mp-id struct (u/lp i) "state" (u/lp j) (u/lp k)])]
333       (filter some?
334               (mapv (fn [k]
335                       (let [name      (ku/key->struct k)
336                             meta-task (task/gen-meta-task name)
337                             task      (task/assemble meta-task mp-id state-key)
338                             value     (kw task)]
339                         (if (and value (or (= value v) (= :all v)))
340                           {kw value :TaskName name :stm-key k} )))
341                     (st/key->keys "tasks"))))))

344  (defn t-run
345    "Runs the task with the given name (from stm).
346    If only the name is provided, results are stored under
347    `core@test@0@response@0@0`.
348  
349    If `mp-id`, `struct`, `i`, `j` and `k` is given, the results are
350    written to `<mp-id@<struct>@<i>@response@<j>@<k>`.  A listener at
351    this key triggers a `cb!` which de-registers and closes the
352    listener. The callback also gets the value of the
353    key (`<mp-id@<struct>@<i>@response@<j>@<k>`) and pretty prints it.
354  
355    REVIEW function is way to large
356    
357    Example:
358    ```clojure
359    (t-run \"DKM_PPC4_DMM-read_temp\")
360    ;;
361    ;; {:t_start 1588071759882,
362    ;; :t_stop 1588071768996,
363    ;; :Result
364    ;; [{:Type dkmppc4,
365    ;;  :Value 24.297828639,
366    ;;  :Unit C,
367    ;;  :SdValue 0.0013625169107,
368    ;;  :N 10}]}
369    ```
370  
371    Debug:
372    ```clojure
373    @st/listeners
374    (st/de-register! \"core\" \"test\" 0 \"response\")
375    ```
376    "
377    ([t]
378     (t-run t "core" "test" 0 0 0))
379    ([t mp-id]
380     (t-run t mp-id "test" 0 0 0))
381    ([t mp-id struct no-idx seq-idx par-idx]
382     (let [no-idx    (u/lp no-idx)
383           seq-idx   (u/lp seq-idx)
384           par-idx   (u/lp par-idx)
385           func       "response"
386           state-key  (u/vec->key [mp-id struct no-idx "state" seq-idx par-idx])
387           resp-key   (u/vec->key [mp-id struct no-idx func    seq-idx par-idx])
388           meta-task  (task/gen-meta-task t)
389           task       (task/assemble meta-task mp-id state-key)]
390       (when (task/dev-action? task)
391         (println "task dispached, wait for response...")
392         (st/register! mp-id struct no-idx func (fn [msg]
393                                                  (when-let [result-key (st/msg->key msg)]
394                                                    (st/de-register! mp-id struct no-idx func)
395                                                    (st/key->val result-key)))))
396       (w/check task))))
