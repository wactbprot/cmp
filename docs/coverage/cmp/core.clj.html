276  (defn t-table
277    "Prints a table of **assembled tasks** stored in
278    **short term memory**. If a `kw` and `val` are given
279    it is used as a filter
280  
281    Example:
282    ```clojure
283    (t-table)
284    ;; same as
285    (t-table :Action :all)
286    ;;
287    (t-table :Action \"TCP\")
288    ;;
289    (t-table :Port \"23\")
290    ```
291    . "
292    ([]
293     (t-table  :Action :all))
294    ([kw v]
295     (pp/print-table
296      (filter some?
297              (into []
298                    (map (fn [k]
299                           (let [name  (st/key->struct k)
300                                 task  (tsk/assemble (tsk/gen-meta-task name))
301                                 value (kw task)]
302                             (if (and value (or (= value v) (= :all v)))
303                               {:stm-key k :Name name kw value} )))
304                         (st/key->keys "tasks")))))))
305  (defn t-run
306    "Runs the task with the given name (from stm).
307    If only the name is provided, results are stored
308    under  `core@test@0@response@0@0`.
309  
310    If  `mp-id`, `struct`, `i`, `j` and  `k` is given,
311    the results are written to `<mp-id@<struct>@<i>@response@<j>@<k>`.
312    A listener at this key triggers a `callback` which de-registers
313    and closes the listener. The callback also gets the value of 
314    the key (`<mp-id@<struct>@<i>@response@<j>@<k>`) and pretty
315    prints it.
316    
317    Example:
318    ```clojure
319    (t-run \"DKM_PPC4_DMM-read_temp\")
320    ;;
321    ;; {:t_start 1588071759882,
322    ;; :t_stop 1588071768996,
323    ;; :Result
324    ;; [{:Type dkmppc4,
325    ;;  :Value 24.297828639,
326    ;;  :Unit C,
327    ;;  :SdValue 0.0013625169107,
328    ;;  :N 10}]}
329    ```  
330    "
331    ([name]
332     (t-run name "core" "test" 0 0 0))
333    ([name mp-id struct i j k]
334     (let [func      "response"
335           state-key (u/vec->key[mp-id struct i "state" j k])
336           resp-key  (u/vec->key[mp-id struct i func j k])
337           meta-task (assoc (tsk/gen-meta-task name)
338                            :MpName mp-id
339                            :StateKey state-key)
340           callback  (fn [msg]
341                       (if-let [k (st/msg->key msg)]
342                         (do
343                           (st/de-register! mp-id struct i func)
344                           (pp/pprint (st/key->val k)))))]
345       (st/register! mp-id struct i func callback)
346       (work/dispatch! (tsk/assemble meta-task) state-key)
347       (timbre/info "task dispached, wait for response"))))
