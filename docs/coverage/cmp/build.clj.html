011  (defn store-exchange
012    "Stores the exchange data."
013    [p {exchange :Exchange}]
014    (doseq [[k v] exchange]
015      (st/set-val! (st/exch-path p (name k)) v)))

019  ;;------------------------------
020  
021  (defn store-defin
022    "Stores the definition section."
023    [p idx defin]
024    (doall
025     (map-indexed
026      (fn [jdx s]
027        (doall
028         (map-indexed
029          (fn [kdx ptsk]
030            (st/set-val! (st/cont-defin-path p idx jdx kdx)  ptsk)
031            (st/set-val! (st/cont-state-path p idx jdx kdx) "ready"))
032          s)))
033      defin)))
034  
035  (defn store-container
036    "Stores a single container"
037    [p idx {descr :Description
038            title :Title
039            ctrl  :Ctrl
040            elem  :Element
041            defin :Definition}]           
042    (st/set-val! (st/cont-title-path p idx) title)
043    (st/set-val! (st/cont-descr-path p idx) descr)
044    (st/set-val! (st/cont-ctrl-path p idx) ctrl)
045    (st/set-val! (st/cont-elem-path p idx) elem)
046    (store-defin p idx defin))

048  (defn store-all-container
049    "Triggers the storing of the singel containers"
050    [p {conts :Container}]
051    (doall
052     (map-indexed
053      (fn [idx cont] (store-container p idx cont))
054      conts)))

082  (defn store-definitions
083    "Stores a definition given in the definition section
084    (second way beside container to provide definitions).
085    This includes `DefinitionClass` and `Conditions`."
086    [p idx ds]
087    (let [{cls :DefinitionClass
088           descr :ShortDescr
089           conds :Condition
090           defin :Definition} ds]
091      (st/set-val! (st/defins-descr-path p idx) descr)
092      (st/set-val! (st/defins-class-path p idx) cls)
093      (store-conds p idx conds)
094      (store-defins p idx defin)
095      (st/set-val! (st/defins-ctrl-path p idx) "ready")))

100    (doall
101     (map-indexed
102      (fn [idx ds] (store-definitions p idx ds))
103      defins)))

124  (defn store
125    "Triggers the storing of `meta`, `exchange`,
126    `container`s etc. to the short term memory.
127    Clears up the fields before. 
128    ```clojure
129    ;; use metadata example input
130    (store ((meta (var store)) :example-input))
131    ```"
132    {:example-input (read-string
133                     (slurp "resources/mpd-ref.edn"))}
134    [{id :_id rev :_rev mp :Mp}]
135    (let [p (u/main-path id)]
136      (st/clear (st/meta-prefix p))
137      (st/clear (st/exch-prefix p))
138      (st/clear (st/cont-prefix p))
139      (st/clear (st/defins-prefix p))
140      (store-meta p mp)
141      (store-exchange p mp)
142      (store-all-container p mp)
143      (store-all-definitions p mp)))

145  (defn store-task
146    "Stores the given `task` unter the
147    path `tasks@<TaskName>`."
148    [task]
149    (st/set-val!
150     (u/vec->key ["tasks" (:TaskName task)])
151     (u/doc->safe-doc task)))
