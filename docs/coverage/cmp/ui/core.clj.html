109  (defmethod td-value :default
110    [m kw]
111    (if-let [x (kw m)]
112      (cond
113        (boolean? x) [:div {:class "tag"} x]
114        (number? x)  [:div {:class "tag"} x]
115        (string? x)  [:div {:class "tag" :id (make-selectable (:key m))} x]
116        (map?    x)  (into [:ul] (mapv (fn [[k v]] [:li [:span (td-value v k)]]) x)))
117      [:span  {:class "tag"} "::"]))

124  (defn t-head
125    [conf kws]
126    (into
127     (into [:thead] (mapv (fn [x] [:col {:class (name x)}]) kws))
128     (mapv (fn [x] [:th (or (get-in conf [:ui :trans x]) x)]) kws)))

130  (defn td [m kws] (mapv (fn [kw] [:td (td-value m kw)]) kws))
131  
132  (defn t-row [m kws] (mapv (fn [x] (into [:tr ] (td x kws))) m))
133  
134  (defn t-base [conf kws]
135    [:table {:class "table is-hoverable is-fullwidth fixed"}

138  (defn table
139    ([conf data]
140     (table conf data (kw-head data))) 
141    ([conf data head]
142     (if (empty? data) (empty-msg "no table data")
143         (into (t-base conf head) (t-row data head)))))
