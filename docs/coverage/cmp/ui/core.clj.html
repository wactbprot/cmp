001  (ns cmp.ui.core
002    (:require
003       [hiccup.form :as hf]
004       [hiccup.page :as hp]))
005  
006  (defn empty-msg [s] [:span {:class "tag is-info"} s])
007  
008  ;;------------------------------
009  ;; table funs
010  ;;------------------------------
011  (defn kw-head [m] (keys (first m)))
012  
013  (defn t-head  [kws] (into [:thead] (mapv (fn [x] [:th x]) kws))) 
014  
015  (defn td      [m kws] (mapv (fn [kw] [:td (kw m)]) kws))
016  
017  (defn t-row   [m kws] (mapv (fn [x]  (into [:tr] (td x kws))) m))
018  
019  (defn t-base  [kws]   [:table {:class "table"} (t-head kws)])
020  
021  (defn table
022    [conf data]
023    (if (empty? data) (empty-msg "no table data")
024        (let [h (kw-head data)]
025          (into (t-base h) (t-row data h)))))
026  
027  ;;------------------------------
028  ;; page funs
029  ;;------------------------------
030  (defn page-header
031    [conf]
032    [:head
033     [:title (:page-title conf)]
034     (hp/include-css "/css/bulma.css")
035     (hp/include-css "/css/all.css")])
036  
037  (defn index-title
038    [conf]
039    [:section {:class "hero is-info"}
040     [:div {:class "hero-body"}
041        [:div {:class "container"}
042         [:h1 {:class "title"} (:main-title conf)]
043         [:h2 {:class "subtitle"} "###"]]]])
044  
045  (defn index
046    [conf body]
047    (hp/html5
048     (page-header conf)
049     [:body
050      (index-title conf)
051      [:section {:class "section"}
052       [:div {:class "container content"}
053        [:div {:class "box"}
054         [:div {:class "columns"}
055          body]]]]
056      (hp/include-js "/js/jquery-3.5.1.min.js")
057      (hp/include-js "/js/ws.js")
058      (hp/include-js "/js/main.js")]))
