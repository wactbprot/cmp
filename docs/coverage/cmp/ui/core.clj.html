119  (defmethod td-value :default
120    [m kw]
121    (if-let [x (kw m)]
122      (cond
123        (boolean? x) [:div {:class "tag"} x]
124        (number? x)  [:div {:class "tag"} x]
125        (string? x)  [:div {:class (str "is-size-6 "x) :id (make-selectable (:key m))} x]
126        (map?    x)  (into [:ul] (mapv (fn [[k v]] [:li [:span (td-value v k)]]) x)))
127      [:span  {:class "tag"} "::"]))
