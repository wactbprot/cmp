040  (defmethod td-value :default
041    [m kw]
042    (if-let [x (kw m)]
043      (cond
044        (boolean? x) [:b x]
045        (string? x)  [:i {:id (:key m)} x]
046        (map?    x)  (into [:ul] (mapv (fn [[k v]] [:li [:span (td-value v k)]]) x)))
047      [:span  {:class "tag"} "::"]))

054  (defn t-head  [kws] (into [:thead] (mapv (fn [x] [:th x]) kws)))

056  (defn td      [m kws] (mapv (fn [kw] [:td (td-value m kw)]) kws))

058  (defn t-row   [m kws] (mapv (fn [x] (into [:tr] (td x kws))) m))

060  (defn t-base  [kws]
061    [:table {:class "table is-hoverable is-fullwidth"}

064  (defn table
065    [conf data]
066    (if (empty? data) (empty-msg "no table data")
067        (let [h (kw-head data)]
068          (into (t-base h) (t-row data h)))))
