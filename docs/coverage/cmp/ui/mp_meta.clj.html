008  (defn doc-link
009    [conf m]
010    (when-let [id  (:doc-id m)]
011      [:p  [:a {:href (str (config/lt-url conf) "/_utils/#database/"
012                           (:lt-db conf) "/" id) }
013            [:b  id ]] [:span {:class "tag is-info"} (:doc-version m)]]))
014  
015  (defn card-content
016    [conf m]
017    [:div {:class "card-content"}  
018     [:div {:class "columns"}
019      [:div {:class "column"}
020       [:figure {:class "image is-3by1"}
021        (ui/img conf m "../../")]]
022      [:div {:class "column"}
023       (ui/mp-id-link conf m)
024       [:p [:b "Standard: "          (:std m)]]
025       [:p [:b "Beschreibung: "] [:i (:descr m)]]]]
026     (when-not (empty? (:docs m))
027       (into [:p [:b "CAL-Dokumente: "]]
028             (mapv (fn [d]  (doc-link conf d)) (:docs m))))])
029    
030  (defn card [conf m]
031    (ui/card-template conf m (card-content conf m) (ui/card-footer conf (assoc m :no-idx (u/lp 0)))))
032  
033  (defn view [conf req data] (ui/index conf req (card conf data)))
