041  (defn button
042    [m kw cls]
043    (when-let [value (kw m)]
044      (let [ds "button is-small setter "
045            cs (condp = cls
046                 :info    "is-outlined is-light is-info"
047                 :success "is-outlined is-light is-success"
048                 :warn    "is-outlined is-light is-warning"
049                 :error   "is-outlined is-light is-danger"
050                 "is-light is-outlined is-primary")]
051        [:button {:class (str ds cs)
052                  :data-value (kw m)
053                  :data-url (post-url m)
054                  :data-key (:key m)}
055         value])))
