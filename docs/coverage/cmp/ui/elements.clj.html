001  (ns cmp.ui.elements
002    (:require
003     [cmp.config        :as config]
004     [cmp.ui.index      :as ui]
005     [cmp.handler-utils :as au]
006     [cmp.st-utils      :as stu]
007     [cmp.utils         :as u]))
008  
009  (defn label
010    [conf k s]
011    [:div {:class "field-label"}
012     [:label {:class "label"} s]])
013  
014  (defn input
015    [conf k g x]
016    [:div {:class "field-body"}
017      [:div {:class "field"}
018       [:p {:class "control"}
019        [:input {:class "input is-info setter"  :value x}]]]])
020  
021  (defn type-unit-value
022  [conf k g m]
023    [:div {:class "field is-horizontal"}
024     (label conf k "Type") (input conf k g (:Type m))
025     (label conf k "Value")(input conf k g (:Value m))
026     (label conf k "Unit") (input conf k g (:Unit m))])
027  
028  (defn selected-ready
029    [conf k g m]
030    [:div {:class "field is-horizontal"}
031     [:div {:class "select exchange"}
032      (into [:select] 
033            (mapv (fn [e] [:option {:value (:value e)} (:display e)]) (:Select m)))]
034     [:button {:class "button is-info setter"
035               :data-value {:Ready "ok"}
036               :data-url (ui/post-url g)
037               :data-key (stu/exch-key (:mp-id g) k)} "ok"]])
038  
039  (defn elem
040    [conf k g m]
041    (let [ks (keys m)]
042      (cond 
043        (every? #{:Type :Unit :Value} ks)       (type-unit-value  conf k g m)
044        (every? #{:Selected :Select :Ready} ks) (selected-ready   conf k g m)
045        :default [:div k m])))
046  
047  (defn content
048    [conf req data]
049    (into [:section]
050          (mapv
051           (fn [m]
052             (when-not (empty? (:elem-values m))
053               [:div {:class "box"}
054                [:h1 (:title m)]
055                (into [:p]
056                      (mapv (fn [k v]
057                              (elem conf k m v))
058                            (:elem-keys-vec m)
059                            (:elem-values m)))]))
060           data)))
061  
062  (defn view [conf req data] (ui/index conf req (content conf req data)))
