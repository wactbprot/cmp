074    [conf m kw]

076      (cond

078        (number? x)  [:div {:class "tag"} x]
079        (string? x)  [:div {:class (str "is-size-6 " x) :id (ui/make-selectable (:key m))} x]
080        (map?    x)  (into [:ul] (mapv (fn [[k v]] [:li [:span (td-value v k)]]) x)))
081      [:span  {:class "tag"} "::"]))

086  (defn td [conf m kws] (mapv (fn [kw] [:td (td-value conf m kw)]) kws))
087  
088  (defn table-row [conf m kws] (mapv (fn [x] (into [:tr ] (td conf x kws))) m))
089  
090  (defn table-head
091    [conf kws]
092    (into
093     (into [:thead] (mapv (fn [x] [:col {:class (name x)}]) kws))
094     (mapv (fn [x] [:th (or (get-in conf [:ui :trans x]) x)]) kws)))
095  
096  (defn table-base
097    [conf kws]
098    [:table {:class "table is-bordered is-hoverable is-fullwidth"}
099     (table-head conf kws)])
100  
101  (defn table
102    ([conf data]
103     (table conf data (keys (first data)))) 
104    ([conf data head]
105     (if (empty? data) (ui/empty-msg "no table data")
106         (into (table-base conf head) (table-row conf data head)))))

109  ;; modal message
110  ;;------------------------------
111  (defn message-id
112    [conf m]
113    (ui/make-selectable (stu/key->message-key (:state-key m))))
114    
115  (defn message
116    [conf m]
117    [:div {:class "modal" :id (message-id conf m)} 
118     [:div {:class "modal-background"}]
119     [:div {:class "modal-content"}
120      [:div {:class "notification is-light is-info"}]
121      [:button {:class "message_ok button is-light is-success"
122                :data-url (ui/post-url m)
123                :data-key (stu/key->message-key (:state-key m))
124                :data-value "ok"} "ok"]]])
125  
126  (defn messages [conf data]
127    (into [:div] (mapv (fn [m]
128                         (when (= 0 (+ (u/ensure-int (:seq-idx m))
129                                       (u/ensure-int (:par-idx m))))
130                           (message conf m))) data)))

143                 (table conf data cols)
144                 (messages conf data)])))
