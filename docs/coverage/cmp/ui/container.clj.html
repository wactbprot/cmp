069        (number? x)  [:div {:class "tag"} x]

072      [:span  {:class "tag"} "::"]))

077  (defn td [conf m kws] (mapv (fn [kw] [:td (td-value conf m kw)]) kws))
078  
079  (defn table-row [conf m kws] (mapv (fn [x] (into [:tr ] (td conf x kws))) m))
080  
081  (defn table-head
082    [conf kws]
083    (into
084     (into [:thead] (mapv (fn [x] [:col {:class (name x)}]) kws))
085     (mapv (fn [x] [:th (or (get-in conf [:ui :trans x]) x)]) kws)))
086  
087  (defn table-base
088    [conf kws]
089    [:table {:class "table is-hoverable is-fullwidth fixed"}
090     (table-head conf kws)])
091  
092  (defn table
093    ([conf data]
094     (table conf data (keys (first data)))) 
095    ([conf data head]
096     (if (empty? data) (ui/empty-msg "no table data")
097         (into (table-base conf head) (table-row conf data head)))))

110    (ui/index conf req
111              (into [:div]
112                    [[:h3 {:class "title"} (:title (first data))]
113                     (table conf data cols)]))))
