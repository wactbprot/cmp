001  (ns cmp.ui.ws
002    (:require
003     [cheshire.core           :as che]
004     [com.brunobonacci.mulog  :as mu]
005     [org.httpkit.server      :refer [with-channel
006                                      on-receive
007                                      on-close
008                                      send!]]))
009  
010  (defonce ws-clients (atom {}))
011  
012  (defn msg-received
013    [msg]
014    (let [data (che/decode msg)]
015      (mu/log ::msg-received :message "msg/data received")))
016  
017  (defn main
018    [conf req]
019    (with-channel req channel
020      (mu/log ::ws :message "connected")
021      (swap! ws-clients assoc channel true)
022      (on-receive channel #'msg-received)
023      (on-close channel (fn [status]
024                          (swap! ws-clients dissoc channel)
025                          (mu/log ::ws :message "closed, status" :status status)))))
026  
027  (defn send-to-ws-clients
028    [conf m]
029    (doseq [client (keys @ws-clients)]
030      (send! client (che/encode m))))
