001  (ns cmp.pubsubstat
002    ^{:author "wactbprot"
003      :doc "Prints a table wthi the n latest pub sub events."}
004    (:require [cmp.st-mem :as st]
005              [clojure.core.async :as a]
006              [cmp.utils :as u]
007              [clojure.pprint :as pp]))
008  
009  (def table (atom []))
010  
011  (defn cb!
012    [msg]
013    (let [d   (u/get-date-object)
014          k   (st/msg->key msg)
015          val (st/key->val k)]
016      (swap! table conj {:h (u/get-hour d)
017                         :m (u/get-min d)
018                         :s (u/get-sec d)
019                         :meth (nth msg 0)
020                         :k k
021                         :val val })
022      (pp/print-table (deref table))))
023  
024  (defn start
025    "Registers a listener. Pretty prints a table
026    on events."
027    ([]
028     (start "*" "*" "*" "*"))
029    ([mp-id struct]
030     (start mp-id struct "*" "*"))
031    ([mp-id struct i]
032     (start mp-id struct i "*"))
033    ([mp-id struct i func]
034    (st/register! mp-id struct i func cb!)))
035  
036  (defn stop
037    "De-registers the listener. Resets the table `atom`."
038    ([]
039     (stop "*" "*" "*" "*"))
040    ([mp-id struct]
041     (stop mp-id struct "*" "*"))
042    ([mp-id struct i]
043     (stop mp-id struct i "*"))
044    ([mp-id struct i func]
045     (reset! table [])
046     (st/de-register! mp-id struct i func)))
