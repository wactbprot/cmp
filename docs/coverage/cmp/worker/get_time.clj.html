001  (ns cmp.worker.get-time
002    ^{:author "wactbprot"
003      :doc "Worker to create a time stamp entry in documents."}
004    (:require [cmp.doc                 :as doc]
005              [cmp.exchange            :as exch]
006              [com.brunobonacci.mulog  :as mu]
007              [cmp.st-mem              :as st]
008              [cmp.utils               :as u]))
009  
010  (defn get-time!
011    "Generates a timestamp object `{:Type <type> :Value (u/get-time)}` and
012    stores it under `:DocPath`."
013    [{type :Type doc-path :DocPath exch-path :ExchangePath state-key :StateKey mp-id :MpName :as task}]
014    (st/set-state! state-key :working)
015    (let [val  {:Type type :Value (u/get-time)}
016          ret  (doc/store! mp-id [val] doc-path)]
017      (when exch-path (exch/to! mp-id val exch-path))
018      (if (:ok ret)
019        (st/set-state! state-key  :executed  "get time executed")
020        (st/set-state! state-key :error "unexpected return value"))))
