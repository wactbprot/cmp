001  (ns cmp.worker.devhub
002    ^{:author "wactbprot"
003      :doc "The devhub worker."}
004    (:require [clj-http.client :as http]
005              [clojure.core.async :as a]
006              [cmp.config :as cfg]
007              [cmp.resp :as resp]
008              [cmp.st-mem :as st]
009              [cmp.utils :as u]
010              [taoensso.timbre :as log]
011              [cmp.worker.pre-script :as ps]))
012  
013  (def post-header (cfg/post-header (cfg/config)))
014  (def dev-hub-url (cfg/dev-hub-url (cfg/config)))
015  
016  (defn resolve-pre-script
017    "Checks if the task has a `:PreScript` (name of the script to run)
018    and an `:Input` key. If not, `task` is returned."
019    [task]
020    (let [{script-name :PreScript
021           input       :PreInput
022           state-key   :StateKey} task]
023      (if (string? script-name)
024        (if (map? input)
025          (condp = (keyword script-name)
026            :set_valve_pos (ps/set-valve-pos task)
027            :get_valve_pos (ps/get-valve-pos task)
028            (do
029              (log/error "script: " script-name " not implemented")
030              (st/set-state! state-key :error)))
031          task)
032        task)))
033    
034  (defn devhub!
035    "Param is called `pre-task` because some tasks come with a
036    `:PreScript` which has to be executed in order to complete
037    the task (sometimes the `:Value` is computed by the
038    `:PreScript`).
039    
040    ```clojure
041     (devhub! {:Action \"TCP\" :Port 23 :Host \"localhost\" :Value \"Hi!\"})
042    ```"
043    [pre-task]
044    (let [state-key (:StateKey pre-task)]
045      (st/set-state! state-key :working)
046      (if-let [task (resolve-pre-script pre-task)]
047        (let [req  (assoc post-header :body (u/map->json task))
048              url  dev-hub-url]
049          (log/debug "try to send req to: " url)
050          (a/go
051            (try
052              (let [res (http/post url req)]
053                (resp/check res task state-key))
054              (catch Exception e
055                (st/set-state! state-key :error)
056                (log/error "request failed")))))
057        (do 
058          (log/error (str "failed to exec task at: " state-key))
059          (st/set-state! state-key :error)))))
