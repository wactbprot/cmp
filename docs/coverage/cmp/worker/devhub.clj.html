013  (defn devhub!
014    "Sends `:Value` to [devhub](https://wactbprot.github.io/devhub/) which
015    resolves `PreScript` and `PostScript`.
016    
017    ```clojure
018     (devhub! {:Action \"TCP\" :Port 23 :Host \"localhost\" :Value \"Hi!\"})
019    ```"
020    [task]
021    (let [state-key (:StateKey task)]
022      (st/set-state! state-key :working)
023      (let [request-key (ku/key->request-key state-key)
024            json-task   (u/map->json task)
025            req         (assoc (cfg/json-post-header (cfg/config)) :body json-task)
026            url         (cfg/dev-hub-url (cfg/config))]
027        (log/debug "try to send req to: " url)
028        (st/set-val! request-key json-task)
029        (log/debug "stored json task to: " request-key)
030        (try
031          (resp/check (http/post url req) task state-key)
032          (catch Exception e
033            (st/set-state! state-key :error)
034            (log/error "request failed"))))))
