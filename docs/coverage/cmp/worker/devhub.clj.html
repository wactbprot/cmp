014  
015  (def mtp (cfg/min-task-period (cfg/config)))
016  (def post-header (cfg/post-header (cfg/config)))
017  (def dev-hub-url (cfg/dev-hub-url (cfg/config)))
018  
019  
020  (defn resolve-pre-script
021    "Checks if the task has a `:PreScript` (name of the script to run)
022    and an `:Input` key. If not `task` is returned."
023    [task state-key]
024    (if-let [script-name (keyword (:PreScript task))]
025      (if-let [input (:PreInput task)]
026        (condp = script-name
027          :set_valve_pos (ps/set-valve-pos task)
028          :get_valve_pos (ps/get-valve-pos task)
029          (do
030            (log/error "script: " script-name " not implemented")
031            (st/set-val! state-key "error")))
032        task)
033      task))
034  
035  (defn devhub!
036    "Param is called `pre-task` because some tasks come with a
037    `:PreScript` which has to be executed in order to complete
038    the task (sometimes the `:Value` is computed by the
039    `:PreScript`).
040    
041    ```clojure
042    
043     (devhub! ((meta (var devhub!)) :example-task)
044              ((meta (var devhub!)) :example-state-key))
045    ```"
046    [pre-task state-key]
047    (st/set-val! state-key "working")
048    (Thread/sleep mtp)
049  
050    (if-let [task (resolve-pre-script pre-task state-key)]
051      (let [req (assoc post-header :body (u/map->json task))
052            url dev-hub-url]
053        (log/debug "send req to: " url)
054        (a/go
055          (try
056            (let [res (http/post url req)]
057              (resp/check res task state-key))
058            (catch Exception e
059              (st/set-val! state-key "error")
060              (log/error "request failed"))))
061      (do 
062        (log/error (str "failed to build task for: " state-key))
063        (st/set-val! state-key "error")))))
