001  (ns cmp.worker.read-exchange
002    ^{:author "wactbprot"
003      :doc "Reads from exchange interface and 
004            writes to docpath."}
005    (:require [taoensso.timbre :as log]
006              [cmp.st-mem      :as st]
007              [cmp.doc         :as doc]
008              [cmp.exchange    :as exch]))
009  
010  (defn read-exchange!
011    "Reads the `exch-val` from `:ExchangePath` and writes the result
012    to `:DocPath`. The `exch-val` have to be turned into a vector, to fit
013    the `doc/store!` function.
014    
015    ```clojure
016    (read-exchange! {})
017    ```"
018    [task]
019    (let [{doc-path  :DocPath
020           exch-path :ExchangePath
021           mp-id     :MpName
022           state-key :StateKey} task]
023    (st/set-state! state-key :working)
024    (let [exch-val [(exch/read! mp-id exch-path)]
025          res-doc  (doc/store! mp-id exch-val doc-path)]
026      (cond
027        (:error res-doc) (st/set-state! state-key :error)
028        (:ok res-doc)    (st/set-state! state-key (if (exch/stop-if task) :executed :ready) "res doc ok")
029        :unexpected      (st/set-state! state-key :error)))))
