001  (ns cmp.worker.read-exchange
002    ^{:author "wactbprot"
003      :doc "Reads from exchange interface and 
004            writes to docpath."}
005    (:require [taoensso.timbre :as log]
006              [cmp.st-mem :as st]
007              [cmp.doc :as doc]
008              [cmp.exchange :as exch]))
009  
010  (defn read-exchange!
011    "Reads the `exch-val` from `:ExchangePath` and writes the result
012    to `:DocPath`. The `exch-val` have to be turned into a vector, to fit
013    the `doc/store!` function.
014    
015    ```clojure
016    (read-exchange! {})
017    ```"
018    [{doc-path :DocPath exch-path :ExchangePath mp-id :MpName state-key :StateKey}]
019    (st/set-state! state-key :working)
020    (let [exch-val [(exch/read! mp-id exch-path)]
021          res-doc  (doc/store! mp-id exch-val doc-path)]
022      (cond
023        (:error res-doc) (st/set-state! state-key :error)
024        (:ok res-doc)    (st/set-state! state-key :executed)
025        :unexpected      (st/set-state! state-key :error))))
