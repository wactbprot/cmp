001  (ns cmp.worker.write-exchange
002    ^{:author "wactbprot"
003      :doc "wait worker."}
004    (:require [taoensso.timbre :as timbre]
005              [cmp.st-mem :as st]
006              [cmp.exchange :as exch]
007              [cmp.config :as cfg]))
008  
009  (def mtp (cfg/min-task-period (cfg/config)))
010  
011  (defn write-exchange!
012    "Writes the `:Value` to the exchange.
013    
014    ```clojure
015    (wait! {:WaitTime 1000} \"testpath\")
016    ```"
017    [{val :Value mp-id :MpName state-key :StateKey}]
018  
019    (st/set-val! state-key "working")
020    (timbre/debug "start with write-exchange, already set " state-key  " working")
021    (Thread/sleep mtp)
022  
023    (let [res (exch/to! mp-id val)]
024      (cond
025        (:error res) (let [err-msg (str "error on attempt to write exchange at: "
026                                        state-key)]
027                       (timbre/error err-msg)
028                       (st/set-val! state-key "error")
029                       {:error err-msg})
030        (:ok res)    (let [msg "wrote to exchange"]
031                       (timbre/info msg)
032                       (st/set-val! state-key "executed")
033                       {:ok true})
034        :default (timbre/warn "should not happen"))))
