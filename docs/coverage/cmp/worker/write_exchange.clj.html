001  (ns cmp.worker.write-exchange
002    ^{:author "wactbprot"
003      :doc "wait worker."}
004    (:require [taoensso.timbre :as log]
005              [cmp.st-mem :as st]
006              [cmp.exchange :as exch]
007              [cmp.config :as cfg]))
008  
009  (defn write-exchange!
010    "Writes the `:Value` to the exchange.
011    
012    ```clojure
013    (write-exchange! {:Value {:a 1 :b 2} :MpName \"ref\"})
014    ;;
015    (st/key->val \"ref@exchange@b\")
016    ;; 2
017    ```"
018    [task]
019    (let [{val       :Value
020           mp-id     :MpName
021           state-key :StateKey
022           exch-path :ExchangePath} task]
023    (st/set-state! state-key :working)
024    (let [res (exch/to! mp-id val exch-path)]
025      (cond
026        (:error res) (st/set-state! state-key :error "error on attempt to write exchange")
027        (:ok res)    (st/set-state! state-key (if (exch/stop-if task) :executed :ready) "wrote to exchange")
028        :default     (st/set-state! state-key :error "unclear exchange response")))))
