001  (ns cmp.worker.get-date
002    ^{:author "wactbprot"
003      :doc "Worker to create a date entry in documents."}
004    (:require [cmp.config              :as cfg]
005              [cmp.doc                 :as doc]
006              [cmp.exchange            :as exch]
007              [com.brunobonacci.mulog  :as mu]
008              [cmp.st-mem              :as st]
009              [cmp.utils               :as u]))
010  
011  (defn get-date!
012    "Generates this date object: `[{:Type <type> :Value (u/get-date)}]`
013    and stores it under  `DocPath`.
014  
015    ```clojure
016    (t-assemble \"Common-get_date\")
017    ;; {:TaskName \"Common-get_date\",
018    ;; :Action \"getDate\",
019    ;; :DocPath \"Calibration.Measurement.Date\",
020    ;; :Type \"%type\",
021    ;; :MpName \"core\",
022    ;; :StateKey \"core@test@0@state@0@0\"}
023    ```"
024    [task]
025    (let [{type :Type doc-path :DocPath state-key :StateKey mp-id :MpName} task]
026      (st/set-state! state-key :working)
027      (let [ret (doc/store! mp-id [{:Type type :Value (u/get-date)}] doc-path)]
028        (if (:ok ret)
029          (st/set-state! state-key :executed  "get date executed")
030          (st/set-state! state-key :error "failed to write date")))))
