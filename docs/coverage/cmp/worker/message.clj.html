007              [cmp.config :as cfg]))
008  
009  (defn message!
010    "Writes a message to the exchange.
011    Continues if message is replaced by the string `ok`
012    
013    ```clojure
014    (message! {:Message \"cmp?\" :MpName \"ref\" :StateKey \"ref@container@10@state@0@0\"})
015    ;;
016    (st/key->val \"ref@container@10@message\")
017    ;; cmp?
018    ```"
019    [{msg :Message mp-id :MpName state-key :StateKey}]
020    (st/set-state! state-key :working)
021    (let [struct   (st/key->struct state-key)
022          no-idx   (st/key->no-idx state-key)
023          msg-key  (st/message-path mp-id struct no-idx)
024          func     (st/key->func msg-key)
025          level    "a"
026          callback (fn [_]
027                     (when (contains? u/ok-set (st/key->val msg-key))
028                       (st/set-state! state-key :executed "ready callback for message worker")
029                       (st/de-register! mp-id struct no-idx func level)))]
030      (st/set-val! msg-key msg)
031      (st/register! mp-id struct no-idx func callback level)))
