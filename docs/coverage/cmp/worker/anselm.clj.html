001  (ns cmp.worker.anselm
002    ^{:author "wactbprot"
003      :doc "Worker to interact with a json api."}
004    (:require [cmp.config             :as cfg]
005              [clj-http.client        :as http]
006              [com.brunobonacci.mulog :as mu]
007              [cmp.resp               :as resp]
008              [cmp.st-mem             :as st]
009              [cmp.st-utils           :as stu]
010              [cmp.utils              :as u]))
011  
012  (defn url
013    "Builds up the url for a `anselm` request."
014    [{path :RequestPath}]
015    (str (cfg/anselm-url (cfg/config)) "/" path))
016    
017  (defn req
018    "Builds up the `req`est map for a `anselm` request."
019    [{action :Action state-key :StateKey value :Value}]
020    (assoc (cfg/json-post-header (cfg/config))
021           :body
022           (u/map->json value)))
023  
024  (defn anselm!
025    "Interacts with a json api.
026    
027    ```clojure
028    ;; leads to a post req:
029     (anselm! {:Action \"Anselm\" :Value {:DocPath \"Calibration.Measurement.Values.Position\",
030                                            :Target_pressure_value 1,
031                                            :Target_pressure_unit \"Pa\"}
032    :RequestPath \"dut_max\"})
033    ```"
034    [task]
035    (let [{value :Value state-key :StateKey} task
036          request-key (stu/key->request-key state-key)]
037      (st/set-state! state-key :working)
038      (st/set-val! request-key task)
039      (mu/log ::anselm! :message "stored request, send request" :key request-key)
040      (if-not value
041        (try ; get
042          (resp/check (http/get (url task)) task state-key)
043          (catch Exception e
044            (st/set-state! state-key :error "get request failed")))
045        (try ; post
046          (resp/check (http/post (url task) (req task)) task state-key)
047          (catch Exception e
048            (st/set-state! state-key :error "post request to failed"))))))
