001  (ns cmp.worker.wait
002    ^{:author "wactbprot"
003      :doc "wait worker."}
004    (:require [taoensso.timbre :as timbre]
005              [cmp.st-mem :as st]
006              [cmp.config :as cfg]))
007  
008  (def mtp (cfg/min-task-period (cfg/config)))
009  
010  (defn wait!
011    "Delays the `mp` for the time given with `:WaitTime`.
012    
013    ```clojure
014    (wait! {:WaitTime 1000})
015    ```"
016    [{wait-time :WaitTime state-key :StateKey}]
017    (when state-key
018      (st/set-val! state-key "working")
019      (timbre/debug "start with wait, already set " state-key  " working"))
020    (let [w (read-string (str wait-time))]
021      (if (< w mtp)
022        (Thread/sleep mtp)
023        (Thread/sleep w))
024      (when state-key
025        (timbre/info "wait time (" w "ms) over for " state-key)
026        (Thread/sleep mtp)
027        (st/set-val! state-key "executed"))))
