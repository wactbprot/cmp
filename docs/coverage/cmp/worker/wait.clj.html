009  (defn wait!
010    "Delays the `mp` for the time given with `:WaitTime`.
011  
012    Example:
013    ```clojure
014    (wait! {:WaitTime 1000})
015    ```"
016    [{wait-time :WaitTime state-key :StateKey}]
017    (st/set-state! state-key :working)
018    (Thread/sleep (u/ensure-int wait-time))
019    (st/set-state! state-key :executed "wait time over"))
