001  (ns cmp.worker.wait
002    ^{:author "wactbprot"
003      :doc "wait worker."}
004    (:require [taoensso.timbre :as log]
005              [cmp.st-mem      :as st]
006              [cmp.exchange    :as exch]
007              [cmp.config      :as cfg]))
008  
009  (def mtp (cfg/min-task-period (cfg/config)))
010  (defn wait!
011    "Delays the `mp` for the time given with `:WaitTime`.
012    
013    ```clojure
014    (wait! {:WaitTime 1000})
015    ```"
016    [task]
017    (let [{wait-time :WaitTime
018           state-key :StateKey} task]
019      (st/set-state! state-key :working)
020      (let [w (read-string (str wait-time))]
021        (if (< w mtp)
022          (Thread/sleep mtp)
023          (Thread/sleep w))
024        (st/set-state! state-key (if (exch/stop-if task) :executed :ready) "wait time over"))))
