001  (ns cmp.worker.run-mp
002    ^{:author "wactbprot"
003      :doc "run-mp worker."}
004    (:require [taoensso.timbre :as log]
005              [cmp.st-mem :as st]
006              [cmp.utils :as u]
007              [cmp.config :as cfg]))
008  
009  (defn exec-index
010    "Registers a level b callback for the `i`th container of the mpd `mp`."
011    [{mp :Mp  i :Container state-k :StateKey}]
012    (let [mp        (u/main-path mp)
013          ctrl-k    (st/cont-ctrl-path mp i)
014          func      "ctrl"
015          struct    "container"
016          level     "b"
017          callback  (fn [msg]
018                      (when (st/msg->key msg)
019                        (condp = (keyword (st/key->val ctrl-k))
020                          :ready (do
021                                   (log/debug "ready callback for" ctrl-k)
022                                   (st/set-state! state-k :executed)
023                                   (log/debug "set" state-k " to executed" )
024                                   (st/de-register! mp struct i func level)
025                                   (log/debug "de-registered" mp struct i func level ))
026                          :error (do
027                                   (log/error "error callback for" ctrl-k)
028                                   (st/set-state! state-k :error))
029                          (log/debug "run callback for" ctrl-k))))]
030      (st/register! mp struct i func callback level)
031      (st/set-state! ctrl-k :run)))
032  
033  (defn exec-title
034    "Searches for the given  `:ContainerTitle`. Extracts the `no-idx`
035    and uses the `exec-index` function to register a callback."
036    [{mp :Mp cont-title :ContainerTitle state-key :StateKey}]
037    (let [mp     (u/main-path mp)
038          ks     (st/pat->keys (st/cont-title-path mp "*" ))
039          title? (fn [k] (= cont-title (st/key->val k)))]
040      (if-let [k (first (filter title? ks))]
041        (exec-index {:Mp mp
042                     :Container (st/key->no-idx k)
043                     :StateKey state-key}) 
044        (do
045          (log/error (str "no container with title: >"cont-title "<"))
046          (st/set-state! state-key :error)))))
047  
048  (defn run-mp!
049    "Runs a certain container of a mpd. Task is marked as executed if all
050    tasks in the container are executed."
051    [task]
052    (let [{cont-title :ContainerTitle
053           cont-index :Container
054           state-key  :StateKey} task]
055      (st/set-state! state-key :working)
056      (cond
057        (not (nil? cont-title)) (exec-title task)
058        (not (nil? cont-index)) (exec-index task)
059        :not-found (st/set-state! state-key :error))))
