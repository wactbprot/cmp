001  (ns cmp.worker.run-mp
002    ^{:author "wactbprot"
003      :doc "run-mp worker."}
004    (:require [taoensso.timbre :as timbre]
005              [cmp.st-mem :as st]
006              [cmp.config :as cfg]))
007  
008  (def mtp (cfg/min-task-period (cfg/config)))
009  (comment
010    (defn exec-index
011      [{mp :Mp mp-id :MpName cont :Container state-key :StateKey}]
012      )
013    
014    (defn run-mp!
015      [task]
016      (let [{cont-title :ContainerTitle
017             cont-index :Container
018           state-key  :StateKey} task]
019        (when state-key
020          (st/set-val! state-key "working")
021          (timbre/debug "start with wait, already set " state-key  " working"))
022        (cond
023          (not (nil? cont-title)) (exec-title task)
024          (not (nil? cont-index)) (exec-index task)
025          :not-found (when state-key
026                     (timbre/error (str "no container title or index at: " state-key))
027                     (st/set-val! state-key "error"))))))
