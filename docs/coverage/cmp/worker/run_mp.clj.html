001  (ns cmp.worker.run-mp
002    ^{:author "wactbprot"
003      :doc "run-mp worker."}
004    (:require [cmp.config              :as cfg]
005              [cmp.key-utils           :as ku]
006              [com.brunobonacci.mulog  :as mu]
007              [cmp.st-mem              :as st]
008              [cmp.utils               :as u]))
009  
010  (defn title->no-idx
011    [mp title]
012    (let [mp     (u/extr-main-path mp)
013          ks     (st/pat->keys (ku/cont-title-key mp "*" ))
014          title? (fn [k] (= title (st/key->val k)))]
015      (ku/key->no-idx (first (filter title? ks)))))
016  
017  
018  (defn exec-index
019    "Registers a level b callback for the `i`th container of the mpd `mp`."
020    [{mp :Mp i :Container state-key :StateKey cmd :Cmd}]
021    (let [mp       (u/extr-main-path mp)
022          cmd      (keyword (or cmd "run"))
023          ctrl-key (ku/cont-ctrl-key mp i)
024          func     "ctrl"
025          struct   "container"
026          level    "b"
027          f        (fn [msg]
028                     (when (st/msg->key msg)
029                        (condp = (keyword (st/key->val ctrl-key))
030                          :ready (do
031                                   (mu/log ::exec-index :message "ready callback for" :key ctrl-key)
032                                   (st/set-state! state-key :executed)
033                                   (mu/log ::exec-index :message "set executed" :key state-key)
034                                   (st/de-register! mp struct i func level)
035                                   (mu/log ::exec-index :message "de-registered"))
036                          :error (do
037                                   (mu/log ::exec-index :error "error callback for" :key ctrl-key)
038                                   (st/set-state! state-key :error))
039                          (mu/log ::exec-index :message "run callback not :ready no :error" :key ctrl-key))))]
040      (st/register! mp struct i func f level)
041      (st/set-state! ctrl-key cmd)))
042  
043  (defn exec-title
044    "Searches for the given  `:ContainerTitle`. Extracts the `no-idx`
045    and uses the `exec-index` function to register a callback."
046    [{mp :Mp cont-title :ContainerTitle state-key :StateKey cmd :Cmd}]
047    (if-let [no-idx (title->no-idx mp cont-title )]
048      (exec-index {:Mp mp :Container no-idx :StateKey state-key :Cmd cmd}) 
049      (st/set-state! state-key :error (str "no container with title: >"cont-title "<"))))
050  
051  (defn run-mp!
052    "Runs a certain container of a `mpd`. `:ContainerTitle` is prefered
053    over `:Container` if both are given. The `task` is marked as
054    `:executed` if all tasks in the container are executed."
055    [task]
056    (let [{title :ContainerTitle index :Container state-key :StateKey} task]
057      (st/set-state! state-key :working)
058      (cond
059        title (exec-title task)
060        index (exec-index task)
061        :not-found (st/set-state! state-key :error "neither title nor index"))))
