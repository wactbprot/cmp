001  (ns cmp.worker.run-mp
002    ^{:author "wactbprot"
003      :doc "run-mp worker."}
004    (:require [taoensso.timbre :as log]
005              [cmp.st-mem :as st]
006              [cmp.utils :as u]
007              [cmp.config :as cfg]))
008  
009  (def mtp (cfg/min-task-period (cfg/config)))
010  
011  (defn exec-index
012    "Registers a callback for the `i`th container of the mpd `mp`."
013    [{mp :Mp  i :Container state-k :StateKey}]
014    (let [ctrl-k (st/cont-ctrl-path mp i)]
015      (st/register! mp "container" i "definition" (st/listener-callback ctrl-k state-k))
016      (st/set-val! ctrl-k "run")))
017  
018  (defn exec-title
019    "Searches for the given  `:ContainerTitle`. Extracts the `no-idx`
020    and uses the `exec-index` function to register a callback."
021    [{mp :Mp cont-title :ContainerTitle state-key :StateKey}]
022    (let [ks (st/pat->keys (u/vec->key [mp "container" "*" "title"]))
023          title? (fn [k] (= cont-title (st/key->val k)))]
024      (if-let [k (first (filter title? ks))]
025        (exec-index {:Mp mp  :Container (st/key->no-idx k) :StateKey state-key}) 
026        (do
027          (log/error "no container with title: " cont-title)
028          (st/set-val! state-key "error")))))
029  
030  (defn run-mp!
031    "Runs a certain container of a mpd. Task is marked as executed if all
032    tasks in the container are executed."
033    [task]
034    (let [{cont-title :ContainerTitle
035           cont-index :Container
036           state-key  :StateKey} task]
037      (when state-key
038        (st/set-val! state-key "working")
039        (log/debug "start with wait, already set " state-key  " working"))
040      (cond
041        (not (nil? cont-title)) (exec-title task)
042        (not (nil? cont-index)) (exec-index task)
043        :not-found (when state-key
044                     (log/error (str "no container title or index at: " state-key))
045                     (st/set-val! state-key "error")))))
