001  (ns cmp.worker.run-mp
002    ^{:author "wactbprot"
003      :doc "run-mp worker."}
004    (:require [taoensso.timbre :as log]
005              [cmp.st-mem :as st]
006              [cmp.utils :as u]
007              [cmp.config :as cfg]))
008  
009  (def mtp (cfg/min-task-period (cfg/config)))
010  
011  (defn exec-index
012    [{mp :Mp mp-id :MpName i :Container state-k :StateKey}]
013    (let [ctrl-k (st/cont-ctrl-path mp i)]
014      (st/register! mp-id "container" i "definition" (st/gen-listener-callback ctrl-k state-k))
015      (st/set-val! ctrl-k "run")))
016  
017  
018  (defn exec-title
019    [{mp :Mp mp-id :MpName cont :ContainerTitle state-key :StateKey}]
020    )
021  
022  (defn run-mp!
023      [task]
024      (let [{cont-title :ContainerTitle
025             cont-index :Container
026           state-key  :StateKey} task]
027        (when state-key
028          (st/set-val! state-key "working")
029          (log/debug "start with wait, already set " state-key  " working"))
030        (cond
031          (not (nil? cont-title)) (exec-title task)
032          (not (nil? cont-index)) (exec-index task)
033          :not-found (when state-key
034                       (log/error (str "no container title or index at: " state-key))
035                       (st/set-val! state-key "error")))))
