062  (defn set-valve-pos
063    "In order to avoid starting up a nodeserver to
064    execute the following pre script, this is a clojure
065    translation.
066  
067    To make this working a `Input` like:
068  
069    ```javascript
070    Input: {
071           should:%should,
072           valve: %valve,
073           stateblock1:%stateblock1,
074           stateblock2:%stateblock2,
075           stateblock3:%stateblock3,
076           stateblock4:%stateblock4
077          }
078    ```
079  
080    has to be added to the task.
081  
082    ```javascript
083     var valvestates = {};,
084     if(%stateblock1) valvestates['1'] = %stateblock1;
085     if(%stateblock2) valvestates['2'] = %stateblock2;
086     if(%stateblock3) valvestates['3'] = %stateblock3;
087     if(%stateblock4) valvestates['4'] = %stateblock4;
088     var block =  {'V1': '1', 'V2': '1','V3': '1','V4': 1, 
089     'V5': 2, 'V6': 2, 'V7': 2, 'V8': 2, 
090     'V9': 3, 'V10': 3, 'V11': 3, 'V12': 3, 
091     'V17': 4, 'V18': 4, 'V19': 4, 'V20': 4};
092     var vs = valvestates[block['%valve']];
093     var vpos = {'V1':0, 'V2':2, 'V3':4, 'V4':6,
094                 'V5':0, 'V6':2, 'V7':4, 'V8':6,
095                 'V9':4, 'V10':2, 'V11':0, 'V12':6,
096                 'V17':0, 'V18':1, 'V19':2, 'V20':3,
097     };
098     var wa = {'V1':40003, 'V2':40003, 'V3':40003, 'V4': 40003,
099               'V5':40004,'V6':40004,'V7':40004,'V8':40004,
100               'V9':40005,'V10':40005,'V11':40005,'V12':40005,
101               'V17':40007,'V18':40007,'V19':40007,'V20':40007};
