001  (ns cmp.worker.select
002    ^{:author "wactbprot"
003      :doc "Worker selects a definition from the same `mp-id` 
004            by evaluating the related conditions.
005           "}
006    (:require [taoensso.timbre :as timbre]
007              [clojure.string :as string]
008              [cmp.st-mem :as st]
009              [cmp.exchange :as exch]
010              [cmp.utils :as u]
011              [cmp.config :as cfg]))
012  
013  (def mtp (cfg/min-task-period (cfg/config)))
014  
015  (defn cond-match?
016    "Tests a single condition of the form defined in
017    the `definitions` section.
018    
019    **Example**
020    
021    Condition maps `m` looks like this:
022    
023    ```clojure
024    (u/json->map (st/key->val \"ref@definitions@0@cond@0\"))
025    ;;
026    ;; {
027    ;;  :ExchangePath \"A.Unit\" ; the value under ref@exchange@A/Unit
028    ;;  :Methode \"eq\"          ; should be equal to
029    ;;  :Value \"Pa\"            ; Pa
030    ;; }
031    ```
032    "
033    [k]
034    (let [mp-id (st/key->mp-id k)
035          m     (st/key->val    k)
036          p     (:ExchangePath m)
037          a     (str (exch/comp-val mp-id p))
038          b     (str (:Value m))]
039      (condp = (keyword (:Methode m))
040        :eq (= a b)
041        :lt (< (read-string a) (read-string b))
042        :gt (> (read-string a) (read-string b)))))
043  
044  (defn conds-match?
045    "Gathers all information for the given
046    definitions key comparison and checks
047    all of the found conditions.
048    
049    ```clojure
050    (conds-match? \"ref@definitions@0@class\")
051    ;; false
052    (conds-match? \"ref@definitions@1@class\")
053    ;; true
054  
055    ```
056    "
057    [k]
058    (let [mp-id    (st/key->mp-id k)
059          no-idx   (st/key->no-idx k)
060          k-pat    (u/vec->key [mp-id "definitions" no-idx "cond@*"])
061          cond-ks  (st/pat->keys k-pat)
062          match-ks (filter cond-match? cond-ks)] 
063      (=
064       (count cond-ks)
065       (count match-ks))))
066  
067  (defn start-defins!
068    "Starts the matching `definitions` structure and
069    sets the state of the calling element to `executed`
070    if the `ctrl`  turns to ready (or error if error).
071  
072    "          
073    [match-k state-k]
074    (timbre/debug "start definitions struct " match-k)
075    (let [struct "definitions"
076          func   "ctrl"
077          mp-id  (st/key->mp-id state-k)
078          no-idx (st/key->no-idx match-k)
079          ctrl-k (st/defins-ctrl-path mp-id no-idx)
080          cb!    (fn [msg]
081                   (when-let [k (st/msg->key msg)]
082                     (let [ctrl-val (st/key->val ctrl-k)] 
083                     (condp = ctrl-val
084                       "ready" (do
085                                 (st/set-val! state-k "executed")
086                                 (st/de-register! mp-id struct no-idx func))
087                       "error" (st/key->val ctrl-k)) (st/set-val! state-k "error"))))]
088      (st/register! mp-id struct no-idx func cb!)
089      (st/set-val! ctrl-k "run")))
090  
091  
092  (defn select-definition!
093    "Selects and runs a `Definition` from the `Definitions`
094    section of the current `mp`. 
095    
096    ```clojure
097    (select-definition! {:Action select
098                         :TaskName Common-select_definition,
099                         :DefinitionClass wait} \"teststate\")
100    ```
101    If more than one than one of the definitions condition match
102    the first is used:
103  
104    ```clojure
105    (first (filter conds-match? match-ks))
106    ;; ref@definitions@1@class
107    ```" 
108    [task state-k]
109    (st/set-val! state-k "working")
110    (Thread/sleep mtp)
111    (timbre/debug "start with select, already set " state-k " working")
112    (let [mp-id     (st/key->mp-id state-k)
113          def-cls   (task :DefinitionClass)
114          def-pat   (u/vec->key [mp-id "definitions" "*" "class"])
115          match-ks  (sort
116                     (st/filter-keys-where-val def-pat def-cls))]
117      (if-let [match-k (first (filter conds-match? match-ks))]
118        (start-defins! match-k state-k)
119        (timbre/error "nothing match"))))
