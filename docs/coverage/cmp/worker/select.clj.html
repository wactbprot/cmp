015  (defn cond-match?
016    "Tests a single condition of the form defined in
017    the `definitions` section.
018  
019    Example:
020    ```clojure
021    ```
022    "
023    [l m r]
024    (condp = m
025        :eq (= l r)
026        :lt (< (read-string (str l)) (read-string (str r)))
027        :gt (> (read-string (str l)) (read-string (str r)))))

029  (defn start-defins!
030    "Starts the matching `definitions` structure. `register`s
031    a level b callback. Sets the state of the calling element to `executed`
032    if the `ctrl`  turns to ready (or error if error).
033  
034    "          
035    [match-k state-k]
036    (log/debug "start definitions struct " match-k)
037    (let [struct   "definitions"
038          func     "ctrl"
039          mp-id    (st/key->mp-id state-k)
040          no-idx   (st/key->no-idx match-k)
041          ctrl-k   (st/defins-ctrl-path mp-id no-idx)
042          level    "b"
043          callback  (fn [msg]
044                      (condp = (keyword (st/key->val ctrl-k))
045                        :run   (log/debug "run callback for" ctrl-k)
046                        :ready (do
047                                 (log/debug "ready callback for" ctrl-k)
048                                 (when state-k
049                                   (Thread/sleep mtp)
050                                   (st/set-val! state-k "executed")) 
051                                 (st/de-register! mp-id struct no-idx func level))
052                        :error (do
053                                 (log/error "error callback for" ctrl-k)
054                                 (when state-k
055                                   (Thread/sleep mtp)
056                                   (st/set-val! state-k "error")))))]
057      (st/register! mp-id struct no-idx func callback level)
058      (st/set-val! ctrl-k "run")))

060  (defn cond-key->cond-map
061    "Builds a `cond`ition`-map` belonging to the
062    key  `k`. Replaces the compare value fetched
063    from the exchange interface by means of the
064    `exch/comp-val!`-function.
065  
066    Example:
067    ```clojure
068    (cond-key->cond-map \"ref@definitions@1@cond@0\")
069    ;; {:mp-name \"ref\",
070    ;;  :struct \"definitions\",
071    ;;  :no-idx 1,
072    ;;  :no-jdy 0,
073    ;;  :comp-value \"Pa\",
074    ;;  :meth \"eq\",
075    ;;  :exch-value \"Pa\"}
076  
077    ;; where:
078  
079    (st/key->val \"ref@definitions@1@cond@0\")
080    ;;{:ExchangePath \"A.Unit\", :Methode \"eq\", :Value \"Pa\"}
081  
082    ;; and:
083  
084    (st/key->val \"ref@exchange@A\")
085    ;; {:Unit \"Pa\", :Value 100}
086    ```
087    "
088    [k]
089    (let [mp-name     (st/key->mp-id k)
090          val-map     (st/key->val k)
091          exch-path   (:ExchangePath val-map)
092          left-val    (exch/comp-val! mp-name exch-path)
093          meth        (keyword (:Methode val-map))
094          right-val   (:Value val-map)]
095      {:mp-name     mp-name
096       :struct      (st/key->struct k)
097       :no-idx      (st/key->no-idx k)
098       :no-jdy      (st/key->no-jdx k)
099       :right-value right-val
100       :meth        meth
101       :left-value  left-val
102       :cond-match (cond-match? left-val meth right-val)}))
103  
104  (defn class-key->cond-keys
105    [k]
106    (when k
107      (let [mp-id   (st/key->mp-id k)
108            struct  (st/key->struct k)
109            no-idx  (st/key->no-idx k)
110            cond-k  (st/defins-cond-path mp-id no-idx)]
111        (st/key->keys cond-k))))

113  (defn class->class-keys
114    "Returns the keys where the class is `cls`."
115    [mp-id cls]
116    (let [pat (st/defins-class-path mp-id "*")]
117      (st/filter-keys-where-val pat cls)))

119  (defn select-definition!
120    "Selects and runs a `Definition` from the `Definitions`
121    section of the current `mp`. Builds a `cond`ition`-map`
122    (analog to the `state-map`) in order to avoid the
123    spreading of side effects and easy testing.
124  
125    Example:
126    ```clojure
127    (ns cmp.worker.select)
128    (select-definition! {:MpName ref
129                         :Action select
130                         :TaskName Common-select_definition,
131                         :DefinitionClass wait} )
132    ```
133    If more than one than one of the definitions condition match
134    the first is used:
135  
136    ```clojure
137    (first (filter conds-match? match-ks))
138    ;; ref@definitions@1@class
139    ```" 
140    [{mp-id :MpName cls :DefinitionClass state-key :StateKey}]
141    (st/set-val! state-key "working")
142    (log/debug "start with select, already set " state-key  " working")
143    (Thread/sleep mtp)
144    (let [cond-vec (mapv
145                    (fn [ks]
146                      (mapv cond-key->cond-map ks)) 
147                         (mapv class-key->cond-keys
148                               (class->class-keys mp-id cls)))]
149      cond-vec
150    ))
