013    "Builds a `state-map` by means of the `info-map`.

085  (defn predecessors-executed?
086    "Checks if `all-executed?` in the steps before `i` of `v`."
087    [v i]
088    (let [i (u/ensure-int i)]
089      (if (< 0 i)
090        (every? true? (map
091                       (fn [j]
092                         (all-executed?
093                          (ku/seq-idx->all-par v j)))
094                       (range i)))
095        true)))

130  (defn all-exec!
131    "Handles the case where all `state` interfaces are
132    `\"executed\"`. Gets the value the `ctrl`"
133    [k]
134    (let [ctrl-k   (ku/k->ctrl-k k)
135          cmd      (ctrl-k->cmd ctrl-k)]
136      (log/info "all done at: " k "ctrl interface cmd is: " cmd)
137      (condp = cmd
138        :mon   (do
139                 (de-observe! ctrl-k)
140                 (ready! ctrl-k)
141                 (st/set-val! ctrl-k "mon"))
142        (do (de-observe! ctrl-k)
143            (ready! ctrl-k)
144            (st/set-val! ctrl-k "ready")
145            (log/info "default condp branch in all-exec fn of " k )))))

150  (defn next-map
151    "The `next-map` function returns a map containing the next step to
152    start. See `cmp.state-test/next-map-i` for examples how `next-map`
153    should work.
154  
155    Example:
156    ```clojure
157     (next-map [{:seq-idx 0 :par-idx 0 :state :executed}
158                {:seq-idx 0 :par-idx 1 :state :executed}
159                {:seq-idx 1 :par-idx 0 :state :executed}
160                {:seq-idx 2 :par-idx 0 :state :executed}
161                {:seq-idx 3 :par-idx 0 :state :working}
162                {:seq-idx 3 :par-idx 1 :state :ready}])
163    ;; =>
164    ;; {:seq-idx 3 :par-idx 1 :state :ready}
165    ```
166    "
167    [v]
168    (when-let [next-m (next-ready v)]
169      (when-let [i (:seq-idx next-m)]
170        (when (or (zero? (u/ensure-int i))
171                  (predecessors-executed? v i))
172          next-m))))

177  (defn start-next
178    "Side effect free. Makes [[start-next!]] testable.
179    Gets the state vector `v` and picks the next thing to do.
180    The `ctrl-k`ey is derived from the first map in the
181    the `v`."
182    [v]
183    (let [m      (next-map v)
184          ctrl-k (ku/info-map->ctrl-key (first v))
185          defi-k (ku/info-map->definition-key m)]
186      (cond
187        (errors?       v) {:what :error    :key ctrl-k}
188        (all-executed? v) {:what :all-exec :key ctrl-k}
189        (nil?          m) {:what :nop      :key ctrl-k}
190        :run-worker       {:what :work     :key defi-k})))
191  
192  (defn start-next!
193    "`start-next!` choose the `k` of the upcomming tasks.
194    Then the `worker` set the state to `\"working\"` which triggers the
195    next call to `start-next!`: parallel tasks are started this way.
196  
197    Side effects all around. "
198    [v]
199    (log/debug "call to start next")
200      (when (vector? v)
201        (let [{what :what k :key} (start-next v)]
202          (condp = what
203            :error    (error!     k)
204            :all-exec (all-exec!  k)
205            :nop      (nop!       k)
206            :work     (work/check k)))))

211  (defn observe!
212    "Registers a listener with a [[start-next!]] callback.  Calls
213    `start-next!` as a first trigger.  The register pattern is derived
214    from the key `k` (`ctrl-key`)."
215    [k]
216    (log/info "register start-next! callback and start-next!")
217    (st/register! (st/key->mp-id k) (st/key->struct k) (st/key->no-idx k) "state"
218                  (fn [msg]
219                    (when-let [msg-k (st/msg->key msg)]                   
220                      (log/debug "will call start-next from callback")
221                      (start-next! (ks->state-vec (ku/k->state-ks msg-k))))))
222    (log/debug "will call start-next first trigger")
223    (start-next! (ks->state-vec (ku/k->state-ks k))))

241  (defn dispatch
242    "Dispaches depending on `cmd`."
243    [k cmd]
244    (condp = (keyword cmd)
245      :run     (observe! k)
246      :mon     (observe! k)
247      :stop    (do
248                 (de-observe! k)
249                 (ready! k))
250      :reset   (do
251                 (de-observe! k)
252                 (ready! k))
253      :suspend (de-observe! k)
254      (log/info  "default case state dispach function" )))
