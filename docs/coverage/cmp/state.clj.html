183  (defn find-next
184    "The `find-next` function
185    returns a list of maps containing the next
186    tasks to start. It should work as follows:
187  
188    ```clojure
189    cmp.run>   (def m
190    [{:seq-idx 0, :par-idx 0, :state :ready}
191     {:seq-idx 0, :par-idx 1, :state :ready}
192     {:seq-idx 1, :par-idx 0, :state :ready}
193     {:seq-idx 2, :par-idx 0, :state :ready}
194     {:seq-idx 3, :par-idx 0, :state :ready}
195     {:seq-idx 3, :par-idx 0, :state :ready}])
196    
197    ;;  #'cmp.run/m
198    cmp.run> (find-next m)
199    ;;  {:seq-idx 0, :par-idx 0, :state :ready}
200    cmp.run>   (def m
201    [{:seq-idx 0, :par-idx 0, :state :working}
202     {:seq-idx 0, :par-idx 1, :state :ready}
203     {:seq-idx 1, :par-idx 0, :state :ready}
204     {:seq-idx 2, :par-idx 0, :state :ready}
205     {:seq-idx 3, :par-idx 0, :state :ready}
206     {:seq-idx 3, :par-idx 0, :state :ready}])
207    
208    ;;  #'cmp.run/m
209    cmp.run> (find-next m)
210    ;;  {:seq-idx 0, :par-idx 1, :state :ready}
211    cmp.run>   (def m
212    [{:seq-idx 0, :par-idx 0, :state :working}
213     {:seq-idx 0, :par-idx 1, :state :working}
214     {:seq-idx 1, :par-idx 0, :state :ready}
215     {:seq-idx 2, :par-idx 0, :state :ready}
216     {:seq-idx 3, :par-idx 0, :state :ready}
217     {:seq-idx 3, :par-idx 0, :state :ready}])
218    
219    ;;  #'cmp.run/m
220    cmp.run> (find-next m)
221    ;;  nil
222    cmp.run>   (def m
223    [{:seq-idx 0, :par-idx 0, :state :executed}
224     {:seq-idx 0, :par-idx 1, :state :executed}
225     {:seq-idx 1, :par-idx 0, :state :ready}
226     {:seq-idx 2, :par-idx 0, :state :ready}
227     {:seq-idx 3, :par-idx 0, :state :ready}
228     {:seq-idx 3, :par-idx 0, :state :ready}])
229    
230    ;;  #'cmp.run/m
231    cmp.run> (find-next m)
232    ;;  {:seq-idx 1, :par-idx 0, :state :ready}
233    ```
234    It should not crash on:
235  
236    ```clojure
237    cmp.run> (count (next-ready m))
238    ;; 0
239    cmp.run> (find-next m)
240    ;; nil
241    cmp.run> (find-next {})
242    ;; nil
243    cmp.run> (find-next nil)
244    ;; nil
245    ```"
246    [m]
247    (when-let [next-m (next-ready m)]
248      (when-let [i (next-m :seq-idx)]
249        (when (or
250               (zero? i)
251               (predecessor-executed? m i))
252          next-m))))

266  (defn de-observe!
267    "Opposite of [[observe!]]:
268    De-registers the `state` listener.
269    The de-register pattern is derived
270    from the key  `k` (may be the
271    `ctrl-key` or `state-key`).
272    Resets the state interface afterwards."
273    [k]
274    (st/de-register! (st/key->mp-id k)
275                     (st/key->struct k)
276                     (st/key->no-idx k)
277                     "state")
278    (ready! k))

283  (defn suspend!
284    "Simply de-registers the `state` listener
285    without changing (e.g. ready!) the state interface.
286    The de-register pattern is derived
287    from the key  `k` (may be the
288    `ctrl-key` or `state-key`)."
289    [k]
290    (let [state-ks (k->state-ks k)]
291      (st/de-register! (st/key->mp-id k)
292                       (st/key->struct k)
293                       (st/key->no-idx k)
294                       "state")))

310  (defn all-exec!
311    "Handles the case where all `state` interfaces
312    are `\"executed\"`. Gets the value  the `ctrl`"
313    [k]
314    (log/info "all done at " k)
315    (let [ctrl-k   (k->ctrl-k k)
316          cmd      (ctrl-k->cmd ctrl-k)
317          state-ks (k->state-ks k)]
318      (log/info "ctrl cmd is: " cmd)
319      (condp = cmd
320        :run (do
321               (de-observe! ctrl-k)
322               (st/set-val! ctrl-k "ready"))
323        :mon (do
324               (de-observe! ctrl-k)
325               (st/set-val! ctrl-k "mon"))
326        (log/info "default condp branch in all-exec fn of " k ))))

331  (defn start-next!
332    "Receives the key `k` and picks the next thing to do.
333    `k` is
334    * the `ctrl-key` on the first run
335    * the `state-key` on the following calls
336    The keys (must be a string) and look like this:
337  
338    ```clojure
339    ;; ctrl-key
340    (start-next! \"se3-calib@container@0@ctrl\")
341  
342    ;; state-key
343    (start-next! \"se3-calib@container@0@state@1@1\")
344  
345    ```
346    
347    **NOTE:**
348  
349    `start-next!` only starts the first of
350    `(find-next state-map)` (the upcomming tasks)
351    since the workers set the state to `\"working\"`
352    which triggers the next call to `start-next!`."
353    [x]
354    (log/info "start-next! on " x)
355    (if-let [k x]
356      (let [ctrl-k   (k->ctrl-k k)
357            state-m  (ks->state-map (k->state-ks ctrl-k))
358            next-m   (find-next state-m)]
359        (log/debug "next map is: " next-m)
360        (cond
361          (errors?       state-m) (error!    ctrl-k)
362          (all-executed? state-m) (all-exec! ctrl-k)
363          (nil?           next-m) (nop!      ctrl-k)
364          :run-worker  (work/check
365                        (state-map->struct-key next-m))))))

370  (defn observe!
371    "Registers a listener with a [[start-next!]] callback.
372    Calls `start-next!` as a first trigger.
373    The register pattern is derived
374    from the key  `k` (`ctrl-key`)."
375    [k]
376    (log/info "register start-next! callback and start-next!")
377    (st/register!  (st/key->mp-id k)
378                   (st/key->struct k)
379                   (st/key->no-idx k)
380                   "state"
381                   (fn [msg] (start-next! (st/msg->key msg))))
382    (start-next! k))

406  (defn dispatch
407    "Dispaches depending on `cmd`."
408    [k cmd]
409    (condp = (keyword cmd)
410      :run     (observe!    k)
411      :mon     (observe!    k)
412      :stop    (de-observe! k)
413      :reset   (ready!      k)
414      :suspend (suspend!    k)
415      (log/info  "default case state dispach function" )))
