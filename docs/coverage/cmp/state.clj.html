161    "Returns a map with the next

190  (defn de-observe!
191    "Opposite of [[observe!]]:
192    De-registers the `state` listener.
193    The de-register pattern is derived
194    from the key  `k` (may be the
195    `ctrl-key` or `state-key`).
196    Resets the state interface afterwards."
197    [k]
198    (st/de-register! (st/key->mp-id k)
199                     (st/key->struct k)
200                     (st/key->no-idx k)
201                     "state")
202    (ready! k))

234  (defn all-exec!
235    "Handles the case where all `state` interfaces
236    are `\"executed\"`. Gets the value  the `ctrl`"
237    [k]
238    (let [ctrl-k   (k->ctrl-k k)
239          cmd      (ctrl-k->cmd ctrl-k)]
240      (log/info "all done at: " k "ctrl interface cmd is: " cmd)
241      (condp = cmd
242        :mon   (do
243                 (de-observe! ctrl-k)
244                 (st/set-val! ctrl-k "mon"))
245        (do (de-observe! ctrl-k)
246            (st/set-val! ctrl-k "ready")
247            (log/info "default condp branch in all-exec fn of " k )))))

253  (defn next-map
254    "The `next-map` function returns a map containing the next
255    step to start. See `cmp.state-test/next-map-i` for examples
256    how  `next-map` should work.
257    
258    ```clojure
259     (next-map [{:seq-idx 0, :par-idx 0, :state :executed}
260                {:seq-idx 0, :par-idx 1, :state :executed}
261                {:seq-idx 1, :par-idx 0, :state :executed}
262                {:seq-idx 2, :par-idx 0, :state :executed}
263                {:seq-idx 3, :par-idx 0, :state :working}
264                {:seq-idx 3, :par-idx 1, :state :ready}])
265    ;; cmp.state> {:seq-idx 3, :par-idx 1, :state :ready}
266    ```"
267    [m]
268    (when-let [next-m (next-ready m)]
269      (when-let [i (:seq-idx next-m)]
270        (when (or
271               (zero? i)
272               (predecessor-executed? m i))
273          next-m))))

275  (defn choose-next
276    "Gets the `state-vec` and picks the next thing to do.
277    The `ctrl-k`ey is derived from the first map in the
278    the `state-vec`."
279    [state-vec]
280    (when (vector? state-vec)
281      (let [next-m (next-map state-vec)
282            ctrl-k (state-map->ctrl-key (first state-vec))
283            defi-k (state-map->definition-key next-m)]
284        (cond
285          (errors?       state-vec) {:what :error    :k ctrl-k}
286          (all-executed? state-vec) {:what :all-exec :k ctrl-k}
287          (nil?          next-m)    {:what :nop      :k ctrl-k}
288          :run-worker               {:what :work     :k defi-k}))))
289  
290  (defn start-next!
291    "`start-next!` choose the `k` of the upcomming tasks.
292    Then the `worker` set the state to `\"working\"`
293    which triggers the next call to `start-next!`:
294    parallel tasks are started this way.
295  
296    Side effects all around. "
297    [state-vec]
298    (when (vector? state-vec)
299      (let [{what :what
300             k    :k} (choose-next state-vec)]
301        (condp = what
302          :error    (error! k)
303          :all-exec (all-exec! k)
304          :nop      (nop! k)
305          :work     (work/check k)))))

310  (defn observe!
311    "Registers a listener with a [[start-next!]] callback.
312    Calls `start-next!` as a first trigger.
313    The register pattern is derived
314    from the key  `k` (`ctrl-key`)."
315    [k]
316    (log/info "register start-next! callback and start-next!")
317    (st/register!  (st/key->mp-id k)
318                   (st/key->struct k)
319                   (st/key->no-idx k)
320                   "state"
321                   (fn [msg] 
322                     (start-next! (ks->state-vec
323                                   (k->state-ks
324                                    (st/msg->key msg))))))
325    (start-next! (ks->state-vec
326                  (k->state-ks k))))

346  (defn dispatch
347    "Dispaches depending on `cmd`."
348    [k cmd]
349    (condp = (keyword cmd)
350      :run     (observe!    k)
351      :mon     (observe!    k)
352      :stop    (de-observe! k)
353      :reset   (ready!      k)
354      :suspend (suspend!    k)
355      (log/info  "default case state dispach function" )))
