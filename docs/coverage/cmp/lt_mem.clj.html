001  (ns cmp.lt-mem
002    (:require [clojure.string :as string]
003              [com.ashafa.clutch :as couch]
004              [cmp.config :as cfg]
005              [taoensso.timbre :as log]))
006  
007  (defn id->doc
008    "Gets a document from the long term memory."
009    [id]
010    (log/debug "try to get document with id: " id)
011    (try
012      (couch/get-document (cfg/lt-conn (cfg/config)) id)
013      (catch Exception e
014        (log/error "catch error on attempt to get doc: " id))))
015  
016  (defn put-doc
017    "Saves a document to the long term memory."
018    [doc]
019    (log/debug "try to save document with id: " (:_id doc))
020    (try
021      (couch/put-document (cfg/lt-conn (cfg/config)) doc)
022      (catch Exception e
023        (log/error "catch error on attempt to put doc"))))
024  
025  (defn all-tasks
026    "Returns all tasks."
027    []
028    (log/debug "get tasks from ltm")
029    (couch/get-view (cfg/lt-conn (cfg/config)) "dbmp" "tasks"))
