001  (ns cmp.lt-mem
002    (:require [cmp.config              :as cfg]
003              [com.ashafa.clutch       :as couch]
004              [com.brunobonacci.mulog  :as mu]
005              [clojure.string          :as string]))
006  
007  (def conn (cfg/lt-conn (cfg/config)))
008  
009  ;;------------------------------
010  ;; get doc
011  ;;------------------------------
012  (defn id->doc
013    "Gets a document from the long term memory."
014    [id]
015    (mu/log ::id->doc :message "try to get document" :doc-id id)
016    (try
017      (couch/get-document conn id)
018      (catch Exception e (mu/log ::id->doc :error (.getMessage e) :doc-id id))))
019  
020  ;;------------------------------
021  ;; put doc
022  ;;------------------------------
023  (defn put-doc
024    "Puts a document to the long term memory."
025    [doc]
026    (mu/log ::id->doc :message "try to put document" :doc-id (:_id doc))
027    (try
028      (couch/put-document conn doc)
029      (catch Exception e (mu/log ::id->doc :error (.getMessage e) :doc-id (:_id doc)))))
030

045  (defn exist?
046    "Returns `true` if a document with the `id` exists.
047  
048    TODO: HEAD request not entire doc
049    
050    Example:
051    ```clojure
052    (exist? \"foo-bar\")
053    ;; =>
054    ;; false
055    ```"
056    [id]
057    (map? (id->doc id)))

059  (defn rev-refresh
060    "Refreshs the revision `_rev` of the document if
061    it exist."
062    [doc]
063    (if-let [db-doc (id->doc (:_id doc))] 
064      (assoc doc :_rev (:_rev db-doc))
065      doc))
