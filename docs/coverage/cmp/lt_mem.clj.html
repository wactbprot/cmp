001  (ns cmp.lt-mem
002    (:require [clojure.core.async :as a]
003              [clojure.string :as string]
004              [com.ashafa.clutch :as couch]
005              [cmp.config :as cfg]
006              [taoensso.timbre :as timbre]))
007  
008  (def conn (cfg/lt-conn (cfg/config)))
009  
010  (defn id->doc
011    "Gets a document from the long term memory."
012    [id]
013    (timbre/debug "try to get document with id: " id)
014    (try
015      (couch/get-document conn id)
016      (catch Exception e
017        (timbre/error "catch error on attempt to get doc: " id))))
018  
019  (defn put-doc
020    "Saves a document to the long term memory."
021    [doc]
022    (timbre/debug "try to save document with id: " (:_id doc))
023    (try
024      (couch/put-document conn doc)
025      (catch Exception e
026        (timbre/error "catch error on attempt to put doc"))))
027  
028  (defn all-tasks
029    "Returns all tasks."
030    []
031    (timbre/debug "get tasks from ltm")
032    (couch/get-view conn "dbmp" "tasks"))
