001  (ns cmp.resp
002    ^{:author "wactbprot"
003      :doc "Catches responses and dispatchs."}
004    (:require [clojure.core.async :as a]
005              [cmp.exchange :as exch]
006              [cmp.excep :as excep]
007              [cmp.doc :as doc]
008              [cmp.lt-mem :as lt]
009              [cmp.st-mem :as st]
010              [cmp.utils :as u]
011              [taoensso.timbre :as timbre]))
012  
013  (defn dispatch
014    "Dispatches responds from outer space.
015    Expected responses are:
016  
017    * Result
018    * ToExchange
019    * error
020    "
021    [body task state-key]
022    (if-let [err (:error body)]
023      (a/>! excep/ch (throw (str "respons: " body " at " state-key)))
024      (let [to-exch  (:ToExchange body)
025            results  (:Result body) 
026            path     (:DocPath body)
027            mp-id    (:MpName task)]
028        (exch/to! mp-id to-exch)
029        (if (and (string? path) (vector? results))
030          (doc/store-results {:place-holder "foo"} results path)
031          ;; here: is Result translated to a vector?
032          ))))
033  
034  
035  
036  ;;------------------------------
037  ;; ctrl channel invoked by run 
038  ;;------------------------------
039  (def ctrl-chan (a/chan))
040  
041  ;;------------------------------
042  ;; ctrl go block 
043  ;;------------------------------
044  (a/go-loop []
045    (let [[res task state-key] (a/<! ctrl-chan)]
046      (timbre/debug "try dispatch response for: " state-key)
047      (try
048        (if-let [status (:status res)]
049          (if-let [body (u/val->clj  (:body res))]
050            (cond
051              (< status 300) (dispatch body task state-key)
052              (= status 304) (dispatch body task state-key)
053              :default (a/>! excep/ch
054                             (throw (str "request for: "
055                                         state-key
056                                         " failed with status: "
057                                         status))))
058            (a/>! excep/ch (throw (str "response body can not be parsed for: "
059                                       state-key))))
060          (a/>! excep/ch (throw (str "no status in header for: "
061                                     state-key))))
062        (catch Exception e
063          (timbre/error "catch error at channel " state-key)
064          (a/>! excep/ch e))))
065    (recur))
