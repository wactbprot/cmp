001  (ns cmp.resp
002    ^{:author "wactbprot"
003      :doc "Catches responses and dispatchs."}
004    (:require [cmp.exchange :as exch]
005              [cmp.doc :as doc]
006              [cmp.lt-mem :as lt]
007              [cmp.st-mem :as st]
008              [cmp.utils :as u]
009              [taoensso.timbre :as log]))
010  
011  (defn dispatch
012    "Dispatches responds from outer space.
013    Expected responses are:
014  
015    * Result
016    * ToExchange
017    * error
018  
019    It's maybe a good idea to save the
020    respons body to a key associated to
021    the state key (done).
022    "
023    [body task state-key]
024    (if-let [err (:error body)]
025      (do
026        (log/error (str "response: " body " at " state-key))
027        (st/set-state! state-key :error))
028      (let [resp-key (st/state-key->response-key state-key)
029            to-exch  (:ToExchange body)
030            results  (:Result body) 
031            doc-path (:DocPath task)
032            mp-id    (:MpName task)]
033        (log/debug "write response body to: " resp-key)
034        (st/set-val! resp-key body)
035        (let [res-exch  (exch/to! mp-id to-exch)
036              res-doc   (doc/store! mp-id results doc-path)]
037          (cond
038            (:error res-exch) (st/set-state! state-key :error)
039            (:error res-doc)  (st/set-state! state-key :error)
040            (and
041             (:ok res-exch)     
042             (:ok res-doc))  (st/set-state! state-key :executed)
043            :unexpected      (st/set-state! state-key :error))))))
044  
045  ;;------------------------------
046  ;; check
047  ;;------------------------------
048  (defn check
049    "Checks a response from outer space.
050    Lookes at the status, parses the body and dispathes."
051    [res task state-key]
052    (if-let [status (:status res)]
053      (if-let [body (u/val->clj (:body res))]
054        (if (< status 400) 
055          (dispatch body task state-key) 
056          (log/error "request for: " state-key" failed with status: " status))            
057        (log/error "body can not be parsed for: " state-key))
058      (log/error "no status in header for: " state-key)))
