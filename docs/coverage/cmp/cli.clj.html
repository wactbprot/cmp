058  ;; build mpd from lt mem
059  ;;------------------------------
060  (defn m-build-ltm
061    "Loads a mpd from long term memory and builds up a `st-mem` version of
062    it. The `mp-id` may be set with [[workon!]]. [[m-start]] is called
063    after mp is build.
064  
065    Example:
066    ```clojure
067    (m-build {} mpid)
068    ```"
069    ([conf mp-id]
070     (m-stop conf mp-id)
071     (st/clear! mp-id)
072     (->> mp-id u/compl-main-path lt/get-doc u/doc->safe-doc build/store)
073     (m-start conf mp-id)))

076  ;; build ref mpd
077  ;;------------------------------
078  (defn m-build-ref
079    "Builds up the `ref`erence mpd provided in `edn` format in the
080    resources folder."
081    [conf]
082    (let [doc (-> (config/ref-mpd conf) slurp read-string)
083          mp-id (u/extr-main-path (:_id doc))]
084      (m-stop conf mp-id)
085      (st/clear! mp-id)
086      (build/store doc)
087      (m-start conf mp-id)))

132  (defn set-ctrl
133    "Writes the command string (`cmd`) to the control interface of a
134    `mpd`. If the `mpd` is already started (see [[m-start]]) the next
135    steps work as follows: `cmd` is written to the short term memory by
136    means of [[cmp.st-mem.set-val!]].  The writing process triggers the
137    `registered` `callback` (registered by [[m-start]]). The `callback`
138    cares about the `cmd`.  `cmd`s are:
139  
140    * `\"run\"`
141    * `\"stop\"`
142    * `\"mon\"`
143    * `\"suspend\"`
144  
145    ```clojure
146    (set-ctrl {} \"ref\" 0\"run\")
147    ```
148  
149    **NOTE:**
150  
151    `set-ctrl` only writes to the `container` structure.  The
152    `definitions` struct should not be started by a
153    user."
154  
155    [conf mp-id i cmd]
156    (st/set-val! (stu/cont-ctrl-key mp-id (u/lp i)) cmd))
