013  (defn kv
014    "Generates a map consisting of `k`ey and `v`alue. The result is
015    merged with `m`."
016    ([k]
017     (kv k nil))
018    ([k m]
019     (merge (assoc (ku/key->info-map k) :value (st/key->val k) :key k) m)))

023  ;;------------------------------
024  (defn listeners
025    "Returns the `reg-key` and the `id` of the running listeners.
026  
027    Example:
028    ```clojure
029    (listener {} {})
030    ```"
031    [conf req]
032    (let [ls @st/listeners]
033      (mapv (fn [k] (assoc (ku/key->reg-map k) :key k :listener-id (get-in ls [k :id]))) (keys ls))))

045    [conf req]

106  (defn set-val!
107    [conf req]
108    (let [k (get-in req [:body :key])
109          v (get-in req [:body :value])]
110      (if (and k v)
111        (if (= "OK" (st/set-val! k v))
112          {:ok true}
113          {:error "on attempt to set value"}) 
114        {:error "missing key or value"})))
