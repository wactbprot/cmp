022  (defn listeners
023    "Returns the `reg-key` and the `id` of the running listeners.
024  
025    Example:
026    ```clojure
027    (listener {} {})
028    ```"
029    [conf req]
030    (let [ls @st/listeners]
031      (mapv (fn [k] (assoc (stu/key->reg-map k) :key k :listener-id (get-in ls [k :id]))) (keys ls))))

062  ;;------------------------------
063  (defn container
064    [conf req]
065    (let [mp-id      (au/req->mp-id req)
066          no-idx     (au/req->no-idx req)
067          seq-idx    (au/req->seq-idx req)
068          state-keys (st/pat->keys (stu/cont-state-key mp-id no-idx seq-idx "*" ))
069          defin-keys (st/pat->keys (stu/cont-defin-key mp-id no-idx seq-idx "*" ))]
070      (mapv (fn [sk dk]
071              (let [no-idx       (stu/key->no-idx sk)
072                    seq-idx      (stu/key->seq-idx sk)
073                    par-idx      (stu/key->par-idx sk)
074                    seq-par-sum (+ (u/ensure-int seq-idx)
075                                    (u/ensure-int par-idx))                  
076                    ctrl-key     (stu/cont-ctrl-key mp-id no-idx)
077                    title-key    (stu/cont-title-key mp-id no-idx)]
078                {:mp-id mp-id
079                 :no-idx no-idx
080                 :seq-idx seq-idx
081                 :par-idx par-idx
082                 :key         sk
083                 :state-key   sk
084                 :ctrl-key   ctrl-key
085                 :run   (when (zero? seq-par-sum) "run")
086                 :mon   (when (zero? seq-par-sum) "mon")
087                 :stop  (when (zero? seq-par-sum) "stop")
088                 :ctrl  (when (zero? seq-par-sum) (st/key->val ctrl-key))
089                 :title (when (zero? seq-par-sum) (st/key->val title-key))
090                 :ready    "ready"
091                 :working  "working"
092                 :executed "executed"
093                 :state    (st/key->val sk)
094                 :task     (tsk/build dk)}))
095            state-keys defin-keys)))

132  (defn set-val!
133    [conf req]
134    (let [k (get-in req [:body :key])
135          v (get-in req [:body :value])]
136      (if (and k v)
137        (if (= "OK" (st/set-val! k v))
138          {:ok true}
139          {:error "on attempt to set value"}) 
140        {:error "missing key or value"})))
