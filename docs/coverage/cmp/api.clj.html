013  (defn kv [k] (assoc (ku/key->info-map k) :value (st/key->val k) :key k))
014  
015  ;;------------------------------
016  ;; listeners 
017  ;;------------------------------
018  (defn listeners
019    "Returns the `reg-key` and the `id` of the running listeners.
020  
021    Example:
022    ```clojure
023    (listener {} {})
024    ```"
025    [conf req]
026    (let [ls @st/listeners]
027      (mapv (fn [k] (assoc (ku/key->reg-map k) :listener-id (get-in ls [k :id]))) (keys ls))))
