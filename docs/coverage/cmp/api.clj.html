040  ;; build mpd from lt mem
041  ;;------------------------------
042  (defn m-build-ltm
043    "Loads a mpd from long term memory and builds up a `st-mem` version of
044    it. The `mp-id` may be set with [[workon!]]. [[m-start]] is called
045    after mp is build.
046  
047    Example:
048    ```clojure
049    (m-build {} mpid)
050    ```"
051    ([conf mp-id]
052     (m-stop conf mp-id)
053     (st/clear! mp-id)
054     (->> mp-id u/compl-main-path lt/get-doc u/doc->safe-doc build/store)
055     (m-start conf mp-id)))

058  ;; build ref mpd
059  ;;------------------------------
060  (defn m-build-ref
061    "Builds up the `ref`erence mpd provided in `edn` format in the
062    resources folder."
063    [conf]
064    (let [doc (-> (config/ref-mpd conf) slurp read-string)
065          mp-id (u/extr-main-path (:_id doc))]
066      (m-stop conf mp-id)
067      (st/clear! mp-id)
068      (build/store doc)
069      (m-start conf mp-id)))

138    [conf req]

140          no-idx  (au/req->no-idx req)]

169  (defn set-val!
170    [conf req]
171    (let [k (get-in req [:body :key])
172          v (get-in req [:body :value])]
173      (if (and k v)
174        (if (= "OK" (st/set-val! k v))
175          {:ok true}
176          {:error "on attempt to set value"}) 
177        {:error "missing key or value"})))
