013  (defn kv [k] {:key k :value (st/key->val k)})
014  
015  (defn listeners
016    "Returns the `reg-key` and the `id` of the running listeners.
017  
018    Example:
019    ```clojure
020    (listener {} {})
021    ```"
022    [conf req]
023    (let [ls @st/listeners]
024      (mapv (fn [k] (assoc (st/reg-map k) :listener-id (get-in ls [k :id]))) (keys ls))))
025  
026  (defn tasks
027    "Returns the `tasks` `key`-`value` pairs available at `st-mem`.
028  
029    Example:
030    ```clojure
031    (tasks {} {})
032    ```"
033    [conf req]
034    (mapv kv (st/key->keys (ku/task-prefix))))
035  
036  (defn container-title
037    "Returns the `container-title` `key`-`value` pairs.
038    
039    Example:
040    ```clojure
041    (container-title {} {} \"ref\")
042    ```"
043    [conf req mp-id]
044    (mapv kv (st/pat->keys (ku/cont-title-key mp-id "*"))))
