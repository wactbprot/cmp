021  ;;------------------------------
022  (defn listeners
023    "Returns the `reg-key` and the `id` of the running listeners.
024  
025    Example:
026    ```clojure
027    (listener {} {})
028    ```"
029    [conf req]
030    (let [ls @st/listeners]
031      (mapv (fn [k] (assoc (ku/key->reg-map k) :key k :listener-id (get-in ls [k :id]))) (keys ls))))

036  (defn tasks
037    "Returns the `tasks` `key`-`value` pairs available at `st-mem`.
038  
039    Example:
040    ```clojure
041    (tasks {} {})
042    ```"
043    [conf req]
044    (mapv au/key-value-map (st/key->keys (ku/task-prefix))))

056       :ndefins (st/key->val (ku/meta-ndefins-key mp-id))

058       :docs    (mapv st/key->val (st/key->keys (ku/id-prefix mp-id)))}))

063  (defn container-ctrl 
064    [conf req]
065    (let [mp-id   (au/req->mp-id req)
066          no-idx  (au/req->no-idx req)]
067      (mapv (fn [ck tk] (au/key-value-map ck {:run   "run"
068                                              :mon   "mon"
069                                              :stop  "stop"
070                                              :title (st/key->val tk)}))
071            (st/pat->keys (ku/cont-ctrl-key mp-id no-idx))
072            (st/pat->keys (ku/cont-title-key mp-id no-idx)))))

074  (defn container-state
075    [conf req]
076    (let [mp-id   (au/req->mp-id req)
077          no-idx  (au/req->no-idx req)]
078      (mapv
079       (fn [k] (au/key-value-map k {:ready    "ready"
080                                    :working  "working"
081                                    :executed "executed"
082                                    :title    (st/key->val (ku/cont-title-key mp-id no-idx))}))
083       (st/pat->keys (ku/cont-state-key mp-id no-idx "*" "*" )))))

087    (let [mp-id   (au/req->mp-id req)

094  ;;------------------------------
095  (defn set-val!
096    [conf req]
097    (let [k (get-in req [:body :key])
098          v (get-in req [:body :value])]
099      (if (and k v)
100        (if (= "OK" (st/set-val! k v))
101          {:ok true}
102          {:error "on attempt to set value"}) 
103        {:error "missing key or value"})))
