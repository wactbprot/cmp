029  (defn get-task
030    "Returns the assembled `task` for the given key `k` related to the
031    `proto-task`. Since the functions in the `cmp.task` namespace are
032    (kept) independent from the tasks position, this info (`:StateKey`
033    holds the position of the task) have to be `assoc`ed
034    (done in `tsk/assemble`)." 
035    [k]
036    (let [state-key (u/replace-key-at-level 3 k "state")]
037      (try (let [proto-task (st/key->val k)
038                 meta-task  (tsk/gen-meta-task proto-task)
039                 mp-id      (ku/key->mp-id k)]
040             (tsk/assemble meta-task mp-id state-key))
041           (catch Exception e
042             (st/set-state! state-key :error (.getMessage e))))))

046  ;;------------------------------
047  (defonce future-reg (atom {}))
048  (defn start!
049    "Starts the worker in a new threat. This means that all workers
050    may be single threated."
051    [worker task]
052    (let [state-key (:StateKey task)]
053      (mu/log ::start! :message "registered worker" :key state-key)
054      (swap! future-reg assoc
055             state-key (future (worker task)))))

059  ;;------------------------------
060  (defn dispatch
061    "Dispatch depending on the `:Action`."
062    [task]
063    (let [action    (keyword (:Action task))
064          state-key (:StateKey task)]
065      (mu/log ::dispatch :message "dispatch task" :key state-key)
066      (condp = action
067        :select         (start! select-definition! task)
068        :runMp          (start! run-mp!            task)
069        :writeExchange  (start! write-exchange!    task)
070        :readExchange   (start! read-exchange!     task)
071        :wait           (start! wait!              task)
072        :getDate        (start! get-date!          task)
073        :getTime        (start! get-time!          task)
074        :message        (start! message!           task)
075        :genDbDoc       (start! gen-db-doc!        task)
076        :replicateDB    (start! replicate!         task)
077        :Anselm         (start! anselm!            task)
078        :MODBUS         (start! devhub!            task)
079        :TCP            (start! devhub!            task)
080        :VXI11          (start! devhub!            task)
081        :EXECUTE        (start! devhub!            task)
082        (st/set-state! state-key :error "No worker for action"))))

087  (defn check
088    "Gets the task. Handles the `:RunIf` and `:StopIf` cases.
089  
090    Example:
091    ```clojure
092    (dispatch {:Action \"wait\" :WaitTime 1000 :StateKey \"testpath\"})
093    ```"  
094    [x]
095    (let [task (if (string? x) (get-task x) x)]
096      (if (exch/run-if task)
097        (dispatch task)
098        (st/set-state! (:StateKey task) :ready "state set by run-if"))))
