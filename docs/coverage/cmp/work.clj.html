002    ^{:author "wactbprot"
003      :doc "Runs the upcomming tasks of a certain container."}
004    (:require [taoensso.timbre :as timbre]

006              [cmp.st-mem :as st]

010              [cmp.worker.devhub :refer [devhub!]]

012              [cmp.task :as tsk]
013              [cmp.utils :as u]))

015  ;;------------------------------
016  ;; task 
017  ;;------------------------------
018  (defn key->task
019    "Returns the assembled `task` for the given key `k`
020    pointing to the `proto-task`.
021    Since the functions in the `cmp.task` namespace are
022    (kept) independent from the tasks position, this info
023    (`:StateKey` holds the position of the task)
024    have to be `assoc`ed here." 
025    [k]
026    (if-let [proto-task (st/key->val k)]
027      (tsk/assemble
028       (assoc (tsk/gen-meta-task proto-task)
029              :MpName    (st/key->mp-id k)
030              :StateKey  (u/replace-key-at-level 3 k "state")))
031      (a/>!! excep/ch (str "No task at: " k))))
032  
033  ;;------------------------------
034  ;; dispatch 
035  ;;------------------------------
036  (defn dispatch!
037    "Dispatches to the workers depending on `(:Action task)`.
038  
039    Since every worker has to manage the state,
040    the  `state-key` is the second parameter.
041  
042    ```clojure
043    (dispatch! {:Action \"wait\" :WaitTime 1000 :StateKey \"testpath\"})
044    ;; #object ... ManyToManyChannel@1247ab05...
045    ;; INFO [cmp.worker.wait:20] - wait time ( 1000 ms) over for  testpath
046    (dispatch! {:Action \"foo\" :StateKey \"testpath\"})
047    ;; ERROR [cmp.work:52] - unknown action:  :foo
048    ```"  
049    [task state-key]
050    (let [action (keyword (:Action task))]
051      (timbre/info "cond for action: " action)
052      (condp = action
053        :wait           (wait!              task state-key)
054        :select         (select-definition! task state-key)
055        :writeExchange  (write-exchange!    task state-key)
056        :MODBUS         (devhub!            task state-key)
057        :TCP            (devhub!            task state-key)
058        :VXI11          (devhub!            task state-key)
059        :EXECUTE        (devhub!            task state-key)
060        (do
061          (timbre/error "unknown action: " action)
062          (st/set-val! state-key "error")))))
063  
064  
065  ;;------------------------------
066  ;; ctrl channel invoked by run 
067  ;;------------------------------
068  (def ctrl-chan (a/chan (a/buffer 10)))
069  
070  ;;------------------------------
071  ;; ctrl go block 
072  ;;------------------------------
073  (a/go-loop []
074    (let [k (a/<! ctrl-chan)]
075      (timbre/info "receive request for: " k)
076      (if-let [task (key->task k)]
077        (if-let [state-key (:StateKey task)]
078          (let [state (st/key->val state-key)] 
079            (if (= state "ready")
080              (do
081                (timbre/debug "try to call worker for: " k)
082                (when (try
083                        (dispatch! task state-key)
084                        (catch Exception e
085                          (timbre/error "catch error on task dispatch for: " k)
086                          (st/set-val! state-key "error")
087                          (a/>! excep/ch e)))))
088              (timbre/debug "state is not ready for: " k)))
089          (timbre/debug "task has no state key: " k))
090        (timbre/debug "no task at: " k)))
091    (recur))
