018  (defn get-task
019    "Returns the assembled `task` for the given key `x` related to the
020    `proto-task` or returns `x` if not a struct key.  Since the
021    functions in the `cmp.task` namespace are (kept) independent from
022    the tasks position, this info (`:StateKey` holds the position of the
023    task) have to be `assoc`ed in `tsk/assemble`." 
024    [x]
025    (if (string? x)
026      (try
027        (if-let [proto-task (st/key->val x)]
028          (let [meta-task (tsk/gen-meta-task proto-task)
029                mp-id     (st/key->mp-id x)
030                state-key (u/replace-key-at-level 3 x "state")]
031            (tsk/assemble meta-task mp-id state-key)))
032        (catch Exception e
033            (str "error at building task at: " x
034                 " catch: " (.getMessage e))))
035      ;; x is already a task:
036      x))
037

039  ;;  dispatch 
040  ;;------------------------------
041  (defn dispatch
042    [task]
043    (let [state-key (:StateKey task)
044          action (keyword (:Action task))]
045      (condp = action
046        :select         (a/go (select-definition! task))
047        :writeExchange  (a/go (write-exchange!    task))
048        :wait           (a/go (wait!              task))
049        :MODBUS         (a/go (devhub!            task))
050        :TCP            (a/go (devhub!            task))
051        :VXI11          (a/go (devhub!            task))
052        :EXECUTE        (a/go (devhub!            task))
053        (when state-key
054          (log/error "unknown action in task:" task)
055          (st/set-val! state-key "error")))))
056  
057  ;;------------------------------
058  ;; check
059  ;;------------------------------
060  (defn check
061    "Gets the task. Handles the `:RunIf` and `:StopIf`.
062  
063    ```clojure
064    (dispatch {:Action \"wait\" :WaitTime 1000 :StateKey \"testpath\"})
065    ```"  
066    [x]
067    (let [task      (get-task  x)
068          state-key (:StateKey task)
069          run-if    (:RunIf    task)
070          stop-if   (:StopIf   task)]
071      (if (nil? run-if)
072        (if (nil? stop-if)
073          (dispatch task) ;; no run-if or stop-if
074          (if (exch/stop-if task)
075            (st/set-val! state-key "executed") ;; stop
076            (dispatch task))) ;; don't stop
077        (if (exch/run-if task)
078          (dispatch task) ;; exec
079          (st/set-val! state-key "ready") ;; don't exec but trigger re-eval
080          ))))
