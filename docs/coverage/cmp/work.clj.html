025  ;;  future registry 
026  ;;------------------------------
027  (defonce future-reg (atom {}))
028  (defn start!
029    "Starts the worker in a new threat. This means that all workers
030    may be single threated."
031    [worker task]
032    (let [state-key (:StateKey task)]
033      (mu/log ::start! :message "registered worker" :key state-key)
034      (swap! future-reg assoc
035             state-key (future (worker task)))))
036  
037  ;;------------------------------
038  ;;  dispatch 
039  ;;------------------------------
040  (defn dispatch
041    "Dispatch depending on the `:Action`."
042    [task]
043    (let [action    (keyword (:Action task))
044          state-key (:StateKey task)]
045      (mu/log ::dispatch :message "dispatch task" :key state-key)
046      (condp = action
047        :select         (start! select-definition! task)
048        :runMp          (start! run-mp!            task)
049        :stopMp         (start! stop-mp!           task)
050        :writeExchange  (start! write-exchange!    task)
051        :readExchange   (start! read-exchange!     task)
052        :wait           (start! wait!              task)
053        :getDate        (start! get-date!          task)
054        :getTime        (start! get-time!          task)
055        :message        (start! message!           task)
056        :genDbDoc       (start! gen-db-doc!        task)
057        :replicateDB    (start! replicate!         task)
058        :Anselm         (start! devproxy!          task)
059        :DevProxy       (start! devproxy!          task)
060        :MODBUS         (start! devhub!            task)
061        :TCP            (start! devhub!            task)
062        :VXI11          (start! devhub!            task)
063        :EXECUTE        (start! devhub!            task)
064        (st/set-state! state-key :error "No worker for action"))))
065  
066  ;;------------------------------
067  ;; check-in
068  ;;------------------------------
069  (defn check
070    "Gets the `task` from the key `k` and calls the `dispach` function on
071    it. Handles the `:RunIf` case. The `:StopIf` case is handeled by the
072    `workers` after processing the task."  
073    [k]
074    (let [k (stu/key->definition-key k)
075          task (tsk/build k)]
076      (if (exch/run-if task)
077        (if (exch/only-if-not task)
078          (dispatch task)
079          (do
080            (Thread/sleep (cfg/stop-if-delay (cfg/config)))
081            (st/set-state! (:StateKey task) :executed "state set by only-if-not")))
082        (do
083          (Thread/sleep (cfg/stop-if-delay (cfg/config)))
084          (st/set-state! (:StateKey task) :ready "state set by run-if")))))
