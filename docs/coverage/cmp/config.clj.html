013  (defn lt-url
014    [c]
015    (let [usr (System/getenv "CAL_USR")
016          pwd (System/getenv "CAL_PWD")]
017      (if (and usr pwd)
018        (str (:lt-prot c)"://"usr":"pwd"@"(:lt-srv c)":"(:lt-port c))
019        (str (:lt-prot c)"://"(:lt-srv c)":"(:lt-port c)))))
