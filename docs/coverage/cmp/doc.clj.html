002    ^{:author "wactbprot"
003      :doc "Handles the documents in which the produced data is stored
004            in.  This may be calibration documents but also measurement
005            docs."}
006    (:require [cmp.lt-mem              :as lt]
007              [cmp.utils               :as u]

013  
014  (defn doc->id [{id :_id}] id)
015  
016  (defn doc->version
017    "Returns the version of the document as an integer value:"
018    [{rev :_rev}]
019    (when-let [v (first (string/split rev  #"-"))] (Integer/parseInt v)))
020  
021  ;;------------------------------
022  ;; extract doc info
023  ;;------------------------------
024  (defn base-info
025    "Returns a map with documents base info."
026    [doc]
027    {:doc-version (doc->version doc) :doc-id (doc->id doc)})
028  
029  (defn doc-type
030    "Returns the type of the document. Assumes the type of the document to
031    be the first key hierarchy beside `:_id` and `:_rev`."
032    [doc m]
033    (first (filter (fn [kw] (not
034                             (or (= :_id kw) (= :_rev kw))))
035                   (keys doc))))
036  
037  (defmulti doc-info
038    "Extracts informations about a document depending on the type."
039    doc-type)
040  
041  (defmethod doc-info :Calibration [doc m] (assoc m :doc-type "Calibration"))
042  (defmethod doc-info :Measurement [doc m] (assoc m :doc-type "Measurement"))
043  (defmethod doc-info :State       [doc m] (assoc m :doc-type "State"))
044  (defmethod doc-info :default     [doc m] (assoc m :doc-type "default"))

047  ;; add
048  ;;------------------------------
049  (defn add
050    "Adds a info map to the short term memory."
051    [mpd-id id]
052    (if-let [doc (lt/get-doc id)]
053      (let [k    (stu/id-key mpd-id id)
054            info (doc-info doc (base-info doc))]
055        (mu/log ::add :message "will add doc info" :doc-id id :key k)
056        (st/set-val! k info))
057      (mu/log ::add :error "no info map added" :doc-id id)))
058  
059  ;;------------------------------
060  ;; rm
061  ;;------------------------------
062  (defn rm
063    "Removes the info map from the short term memory."
064    [mpd-id id]
065    (mu/log ::rm :message "will rm doc info from st-mem" :doc-id id)
066    (st/del-key! (stu/id-key mpd-id id)))

068  ;;------------------------------
069  ;; ids
070  ;;------------------------------
071  (defn ids
072    "Returns the list of ids added.
073  
074    Example:
075    ```clojure
076    (add \"devs\" \"cal-2018-ce3-kk-75003_0002\")
077    ;; hiob DEBUG [cmp.lt-mem:14] - try to get 
078    ;; document with id: cal-2018-ce3-kk-75003_0002
079    ;; OK
080    (ids \"devs\")
081    ;; (cal-2018-ce3-kk-75003_0002)
082    ```"
083    [mp-id]
084    (mapv (fn [k] (stu/key-at-level k 2))
085          (st/key->keys (stu/id-prefix mp-id))))
086  
087  ;;------------------------------
088  ;; renew
089  ;;------------------------------
090  (defn renew
091    "Renew the id interface with the give ids-vector `v`."
092    [mpd-id v]
093    (when (vector? v)
094      (mu/log ::refresh :message "will refresh ids")
095      (mapv (fn [id] (rm mpd-id id)) (ids mpd-id))
096      (mapv (fn [id] (add mpd-id id)) v))
097    {:ok true})
098  
099  ;;------------------------------
100  ;; store with doc-lock
101  ;;------------------------------
102  (def doc-lock (Object.))
103  (defn store!
104    "Stores the `results` vector under the `doc-path` of every document
105    loaded at the given `mp-id`. Checks if the version of each document
106    is `+1`.  Returns `{:ok true}` or `{:error <problem>}`.
107  
108    Example:
109    ```clojure
110    (def results [{:Type \"cmp-test\" :Unit \"Pa\" :Value 1}
111                 {:Type \"cmp-test2\" :Unit \"Pa\" :Value 2}])
112  
113    (def doc-path  \"Calibration.Measurement.Values.Pressure\")  
114  
115    (store! \"ref\" results doc-path)
116    ```"  
117    [mp-id results doc-path]
118    (if (and (string? mp-id) (vector? results) (string? doc-path))
119      (let [ids (ids mp-id)]
120        (if (empty? ids)
121          {:ok true :warn "no documents loaded"}
122          (let [res (map (fn [id]
123                           (locking doc-lock
124                             (mu/log ::store! :message "lock doc" :doc-id id)
125                             (let [in-doc  (lt/get-doc id)
126                                   doc     (insert/store-results in-doc results doc-path)
127                                   out-doc (lt/put-doc doc)]
128                               (mu/log ::store! :message "release lock" :doc-id id))))
129                         ids)]
130            (if-let [n-err (:error (frequencies res))]
131              {:error "got " n-err " during attempt to store results"}
132              {:ok true}))))
133      {:ok true :warn "no doc-path or no mp-id or no results"}))
