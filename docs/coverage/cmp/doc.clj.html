014  (defn doc->version
015    "Returns the version of the document as
016    an integer value:"
017    [{rev :_rev}]
018    (if-let [v (first (string/split rev  #"-"))]
019      (Integer/parseInt v)))

031  (defn base-info
032    "Returns a map with documents base info."
033    [doc]
034    {:doc-version (doc->version doc)
035     :doc-id (doc->id doc)})

037  (defn doc-type
038    "Returns the type of the document. Assumes the
039    type of the document to be the
040    first key hierarchy beside `:_id` and `:_rev`."
041    [doc m]
042    (first
043     (filter (fn [kw] (not
044                (or (= :_id kw) (= :_rev kw))))
045      (keys doc))))

075  (defn add
076    "Adds a info map to the short term memory."
077    [mpd-id id]
078    (if-let [doc (lt/id->doc id)]
079      (let [k    (st/id-path mpd-id id)
080            info (doc-info doc (base-info doc))]
081        (st/set-val! k info))
082      (timbre/error "no doc added")))

095  (defn ids
096    "Returns the list of ids added.
097    ```clojure
098    (add \"devs\" \"cal-2018-ce3-kk-75003_0002\")
099    ;; hiob DEBUG [cmp.lt-mem:14] - try to get 
100    ;; document with id: cal-2018-ce3-kk-75003_0002
101    ;; OK
102    (ids \"devs\")
103    ;; (cal-2018-ce3-kk-75003_0002)
104    ```"
105    [mp-id]
106    (map
107     (fn [k] (u/key-at-level k 2))
108     (st/key->keys
109      (st/id-prefix mp-id))))

114  (defn vector-vals
115    "Ensures that the values behind `:Value`,
116    `:SdValue` and `:N` are vectors."
117    [m]
118    (-> m
119        (u/vector-if :Value)
120        (u/vector-if :SdValue)
121        (u/vector-if :N)))
122  
123  (defn append-and-replace
124    "Append `:Value`, `:SdValue` and `:N` if present.
125    Relaces `:Type` and `:Unit`."  
126    [struct {t :Type v :Value u :Unit n :N s :SdValue}]
127    (->
128     (-> struct
129         (u/replace-if :Type t)
130         (u/replace-if :Unit u))
131     (u/append-if :Value v)
132     (u/append-if :SdValue s)
133     (u/append-if :N n)))
134  
135  (defn fit-in-struct
136    "Fits `m` into the given structure `s`. Function
137    looks up the `:Type` of `m`. If a structure with
138    the same `:Type` exist [[append-and-replace]] is
139    called."
140    [s m]
141    (if-let [t (:Type m)]
142      (let [same-type? (fn [x]  (= (:Type x) t))
143            idx?       (fn [i x] (when (same-type? x) i))]
144            (if-let [idx (first (keep-indexed idx? s))]
145              (assoc s idx (append-and-replace (nth s idx) m))
146              (conj s (vector-vals m))))))
147  
148  (defn store-result
149    "Stores the result map `m` in the given `doc`ument
150    under `p`ath. If `m` contains `:Type` and `:Value` `m`
151    is [[fit-in-struct]] and the structure `s` is assumed
152    to be a `vector`. Other cases (e.g. merge in `:AuxValues`)
153    are straight forward (see [[cmp/test/cmp/doc_test.clj]]
154    for details)."
155    [doc m p]
156    (let [kw-vec (u/path->kw-vec p)]
157      (if (and (:Type m) (:Value m))
158        (if-let [s (get-in doc kw-vec)]
159          (assoc-in doc kw-vec (fit-in-struct s m))
160          (assoc-in doc kw-vec [(vector-vals m)]))
161        (if-let [s (get-in doc kw-vec)]
162          (assoc-in doc kw-vec (merge s m))
163          (assoc-in doc kw-vec m)))))
164  
165  (defn store-results
166    "Takes a vector of maps. Calls `store-result`
167    on each map."
168    [doc res-vec path]
169    (reduce 
170     (fn [doc m] (store-result doc m path))
171     doc res-vec))

173  (defn store!
174    "Stores the `results` vector under
175    the `doc-path` of every document loaded
176    at the given `mp-id`. Checks if the
177    version of each document is `+1`.
178    Returns `{:ok true}` or `{:error <problem>}`.
179  
180    Example:
181    ```clojure
182    (def results [{:Type \"cmp-test\" :Unit \"Pa\" :Value 1}
183                 {:Type \"cmp-test2\" :Unit \"Pa\" :Value 2}])
184  
185    (def doc-path  \"Calibration.Measurement.Values.Pressure\")  
186  
187    (store! \"devs\" results doc-path)
188    ```"  
189    [mp-id results doc-path]
190    (if results
191      (if (and (string? mp-id)
192               (string? doc-path)
193               (vector? results))
194        (let [ids (ids mp-id)]
195          (if (= 0 (count ids))
196            {:ok true :warn "no documents loaded"}
197            (let [res (map
198                       (fn [id]
199                         (let [in-doc  (lt/id->doc id)
200                               doc     (store-results in-doc results doc-path)
201                               out-doc (lt/put-doc doc)]
202                           (if (= (doc->version out-doc)
203                                  (+ 1 (doc->version in-doc)))
204                             :ok
205                             :error)))
206                       ids)]
207              (if-let [n-err (:error (frequencies res))]
208                {:error "got " n-err " during attempt to store results"}
209              {:ok true}))))
210        {:error "wrong store! input"})
211      {:ok true}))
