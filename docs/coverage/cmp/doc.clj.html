002    ^{:author "wactbprot"
003      :doc "Handles the documents in which the 
004            produced data is stored in.

017  (defn base-info
018    "Returns a map with documents base info."
019    [doc]
020    {:doc-version (lt/doc->version doc)
021     :doc-id (lt/doc->id doc)})
022  
023  (defn extr-doc-type
024    "Extracts the document type. Assumes the
025    type of the document to be the
026    first key hierarchy beside `:_id` and `:_rev`."
027    [doc m]
028    (first
029     (filter (fn [kw] (not
030                (or (= :_id kw) (= :_rev kw))))
031      (keys doc))))
032  
033  (defmulti extr-info
034    extr-doc-type)
035  
036  (defmethod extr-info :Calibration
037    [doc m]
038    (assoc m
039           :doc-type "Calibration"))
040  
041  (defmethod extr-info :Measurement
042    [doc m]
043    (assoc m
044           :doc-type "Measurement"))
045  
046  (defmethod extr-info :State
047    [doc m]
048    (assoc m
049           :doc-type "State"))
050  
051  (defmethod extr-info :default
052    [doc m]
053    (assoc m
054           :doc-type "default"))

059  (defn add
060    "Adds a info map to the short term memory."
061    [mpd-id id]
062    (let [path (st/get-id-path mpd-id id)
063          doc  (lt/id->doc id)
064          info (extr-info doc (base-info doc))]
065      (st/set-val! path info)))

070  (defn rm
071    "Removes the info map from the short term memory."
072    [mpd-id id]
073    (st/del-key! (st/get-id-path mpd-id id)))

078  (defn ensure-vector-vals
079    "Ensures that the values behind `:Value`,
080    `:SdValue` and `:N` are vectors."
081    [m]
082    (-> m
083        (u/vector-if :Value)
084        (u/vector-if :SdValue)
085        (u/vector-if :N)))
086  
087  (defn append-and-replace
088    "Append `:Value`, `:SdValue` and `:N` if present.
089    Relaces `:Type` and `:Unit`."  
090    [struct {t :Type v :Value u :Unit n :N s :SdValue}]
091    (->
092     (-> struct
093         (u/replace-if :Type t)
094         (u/replace-if :Unit u))
095     (u/append-if :Value v)
096     (u/append-if :SdValue s)
097     (u/append-if :N n)))
098  
099  (defn fit-in-struct
100    "Fits `m` into the given structure `s`. Function
101    looks up the `:Type` of `m`. If a structure with
102    the same `:Type` exist [[append-and-replace]] is
103    called."
104    [s m]
105    (if-let [t (:Type m)]
106      (let [same-type? (fn [x]  (= (:Type x) t))
107            idx?       (fn [i x] (when (same-type? x) i))]
108            (if-let [idx (first (keep-indexed idx? s))]
109              (assoc s idx (append-and-replace (nth s idx) m))
110              (conj s (ensure-vector-vals m))))))
111  
112  (defn store-result
113    "Stores the result map `m` in the given `doc`ument
114    under `p`ath. If `m` contains `:Type` and `:Value` `m`
115    is [[fit-in-struct]] and the structure `s` is assumed
116    to be a `vector`. Other cases (e.g. merge in `:AuxValues`)
117    are straight forward (see [[cmp/test/cmp/doc_test.clj]]
118    for details)."
119    [doc m p]
120    (let [kw-vec (u/path->kw-vec p)]
121      (if (and (:Type m) (:Value m))
122        (if-let [s (get-in doc kw-vec)]
123          (assoc-in doc kw-vec (fit-in-struct s m))
124          (assoc-in doc kw-vec [(ensure-vector-vals m)]))
125        (if-let [s (get-in doc kw-vec)]
126          (assoc-in doc kw-vec (merge s m))
127          (assoc-in doc kw-vec m)))))
128  
129  (defn store-results
130    "Takes a vector of maps. Calls `store-result`
131    on each map."
132    [doc res-vec path]
133    (reduce 
134     (fn [doc m] (store-result doc m path))
135     doc res-vec))
