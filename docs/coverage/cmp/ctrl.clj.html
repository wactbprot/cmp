001  (ns cmp.ctrl
002    ^{:author "wactbprot"
003      :doc "Observes the `ctrl` interface."}
004    (:require [taoensso.timbre :as timbre]
005              [cmp.st-mem :as st]
006              [clojure.core.async :as a]
007              [cmp.excep :as excep]
008              [cmp.state :as state]
009              [cmp.utils :as u]))
010  
011  ;;------------------------------
012  ;; ctrl-dispatch!
013  ;;------------------------------
014  (defn dispatch
015    "Dispatches on the value of the
016    `ctrl` interface  for the structure
017    belonging to `k`."
018    [k]
019    (timbre/info "ctrl dispatch call for path: " k)
020    (if k
021      (let [cmd (->> k
022                     st/key->val
023                     u/get-next-ctrl)]
024        (a/>!! state/ctrl-chan [k cmd]))))
025  
026  ;;------------------------------
027  ;; stop
028  ;;------------------------------
029  (defn stop
030    "De-registers the listener for the `ctrl`
031    interfacees of the `mp-id`. After stopping
032    the system will no longer react on changes
033    at the `ctrl` interface."
034    [mp-id]
035    (st/de-register! mp-id "*" "*" "ctrl"))
036  
037  ;;------------------------------
038  ;; start
039  ;;------------------------------
040  (defn start
041    "Registers a listener for the `ctrl` interface of
042    the entire `mp-id`. The [[dispatch]] function
043    becomes the listeners `cb!`." 
044    [mp-id]
045    (timbre/info "register ctrl listener for: " mp-id)
046    (let [cb! (fn [msg]
047                     (dispatch
048                      (st/msg->key msg)))]
049     (st/register! mp-id "*" "*" "ctrl" cb!)))
