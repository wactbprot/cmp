001  (ns cmp.ctrl
002    ^{:author "wactbprot"
003      :doc "Observes the `ctrl` interface."}
004    (:require [taoensso.timbre :as timbre]
005              [cmp.st-mem :as st]
006              [clojure.core.async :as a]
007              [cmp.excep :as excep]
008              [cmp.state :as state]
009              [cmp.utils :as u]))
010  
011  
012  ;;------------------------------
013  ;; ctrl-dispatch!
014  ;;------------------------------
015  (defn dispatch
016    "Dispatches on the value of the
017    `ctrl` interface  for the structure
018    belonging to `p`."
019    [p]
020    (timbre/info "ctrl dispatch call for path: " p)
021    (if p
022      (let [cmd (->> p
023                     st/key->val
024                     u/get-next-ctrl)]
025        (a/>!! state/ctrl-chan [p cmd]))))
026  
027  ;;------------------------------
028  ;; stop
029  ;;------------------------------
030  (defn stop
031    "De-registers the listener for the `ctrl`
032    interfacees of the `mp-id`. After stopping
033    the system will no longer react on changes
034    at the `ctrl` interface."
035    [mp-id]
036    (st/de-register! mp-id "*" "*" "ctrl"))
037  
038  ;;------------------------------
039  ;; start
040  ;;------------------------------
041  (defn start
042    "Registers a listener for the `ctrl` interface of
043    the entire `mp-id`. The [[dispatch]] function
044    becomes the listeners `callback`." 
045    [mp-id]
046    (timbre/info "register ctrl listener for: " mp-id)
047    (st/register! mp-id "*" "*" "ctrl"
048                  (fn [msg] (dispatch (st/msg->key msg)))))
049  
050  ;;------------------------------
051  ;; ctrl channel invoked by run 
052  ;;------------------------------
053  (def ctrl-chan (a/chan))
054  
055  ;;------------------------------
056  ;; ctrl go block 
057  ;;------------------------------
058  (a/go-loop []
059    (let [[mp-id cmd] (a/<! ctrl-chan)]
060      (try
061        (timbre/info "receive key: " mp-id " and cmd: " cmd)
062        (condp = (keyword cmd)
063          :start  (start mp-id)
064          :stop   (stop mp-id)
065          (timbre/error "no case for: " mp-id " and cmd: " cmd))
066        (catch Exception e
067          (timbre/error "catch error for: " mp-id)
068          (a/>!! excep/ch e))))
069    (recur))
