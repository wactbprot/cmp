173  (defn clj->val
174    "Casts the given (complex) value `x` to a writable
175    type. `json` is used for complex data types.
176  
177    Example:
178    ```clojure
179    (clj->val {:foo \"bar\"})
180    ;; \"{\"foo\":\"bar\"}\"
181    (clj->val [1 2 3])
182    ;; \"[1,2,3]\"
183    ```
184    "
185    [x]
186    (condp = (class x)
187      clojure.lang.PersistentArrayMap (json/write-str x)
188      clojure.lang.PersistentVector   (json/write-str x)
189      clojure.lang.PersistentHashMap  (json/write-str x)
190      x))
191  
192  (defn clj->str-val
193    [x]
194    (str (clj->val x) ))

199  (defn val->clj
200    "Parses value `v` and returns a
201    clojure type of it.
202  
203    ```clojure
204    (val->clj \"-1e-9\")
205    ;; -1.0E-9
206    ;; class:
207    ;;
208    (class (val->clj \"1.23\"))
209    ;; java.lang.Double
210    (class (val->clj \"a\"))
211    ;; java.lang.String
212    (class (val->clj \"[]\"))
213    ;; clojure.lang.PersistentVector
214    (class (val->clj \"{}\"))
215    ;; clojure.lang.PersistentArrayMap
216    (class (val->clj \"{\"a\":1}\"))
217    ;; clojure.lang.PersistentArrayMap
218    ```
219    "
220    [v]
221    (let [s-pat #"^-?\d+\.?\d*([Ee]\+\d+|[Ee]-\d+|[Ee]\d+)?$"
222          c-pat #"^[\[\{]"]
223      (cond
224        (nil? v) nil
225        (re-find s-pat v) (read-string v)
226        (re-find c-pat v) (json->map v)
227        :else v)))

261  (defn vector-if
262    "Makes the value `v` behind the keyword `kw`
263    a vector if `v` is not nil."
264    [m kw]
265    (if (and (map? m) (keyword? kw))
266      (if-let [v (kw m)]
267        (assoc m kw (val->vec v))
268        m)))

284  (defn append-if
285    "Appends `v` to the value of `k`.
286    If `k` does not exist in `m`, `k [v]` is assoced.
287    If `k` does exist in `m`, `v` is conjed.
288    
289    ```clojure
290    (append-if {:Value [1 2 3]} :Value 4)
291    ;; {:Value [1 2 3 4]}
292    ```"
293    [m k v]
294    (if (and (some? v) (keyword? k))
295      (let [new-v (val->vec v)]
296        (if-let [old-v (k m)]
297          (assoc m k (into [] (concat old-v new-v)))
298          (assoc m k new-v)))
299      m))
