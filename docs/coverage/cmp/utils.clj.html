026  (defn replace-key-at-level
027    "Generates a new key by replacing an old key `k` at the given position
028    `l` with the given string `r`.
029  
030    REVIEW The key levels should have a name or keyword.  Passing
031    integers (`l`) is unimaginative.
032    "
033    [l k r]
034    {:pre [(string? k)
035           (int? l)
036           (string? r)]}
037    (let [v (string/split k re-sep)]
038      (when (< l (count v)) (vec->key (assoc v l r)))))

042    [k l]
043    {:pre [(string? k)
044           (int? l)] }
045    (nth (string/split k re-sep) l nil))

078  (defn ensure-int
079    "Ensures `i` to be integer. Returns 0 as default.
080  
081    ```clojure
082    (u/ensure-int 100)
083    ;; 100
084    (u/ensure-int \"w\")
085    ;; 0
086    (u/ensure-int \"00\")
087    ;; 0
088    (u/ensure-int \"10\")
089    ;; 10
090    (u/ensure-int true)
091    ;; 0
092    ```
093    "
094    [i]
095    (if (integer? i)
096      i
097      (try
098        (Integer/parseInt i)
099        (catch Exception e
100          0))))

102  (defn pad-ok?
103    "Checks if the padding of `i` is ok. `\"*\"` serves pattern matching."
104    ([i]
105     (pad-ok? i (cfg/key-pad-length (cfg/config))))
106    ([i n]
107     (cond
108       (= i "*")         true 
109       (and
110        (string? i)
111        (= n (count i))) true
112       :else             false)))

114  (defn lp
115    "Left pad the given number if it is not a string. Default
116    is `(cfg/key-pad-length (cfg/config))`.
117  
118    Example:
119    ```clojure
120     (u/lp 2)
121    ;; \"002\"
122    (u/lp \"02\")
123    ;; \"002\"
124    (u/lp 2)
125    ;; \"002\"
126    (u/lp true)
127    ;; \"000\"
128    (u/lp \"003\")
129    ;; \"003\"
130    (u/lp \"000003\")
131    ;; \"003\"
132    ```
133    "
134    ([i]
135     (if (pad-ok? i) i (lp (ensure-int i) 3)))
136    ([i n]
137     (if (pad-ok? i) i (format (str "%0" n "d") (ensure-int i)))))

142  (defn main-path
143    "Extracts the main path. Should work on `mpd-aaa-bbb` as well as on
144    `aaa-bbb`.
145  
146    ```clojure
147    (u/main-path \"aa\")
148    ;; \"aa\"
149    (u/main-path \"aa-bbb\")
150    ;; \"aa-bbb\"
151    (u/main-path \"aa-bbb-lll\")
152    ;; \"aa-bbb-lll\"
153    ```
154    "
155    [s]
156    (if (string/starts-with? s "mpd-")
157      (second (re-matches  #"^mpd-([a-z0-3\-_]*)$" s))
158      s))

175  (defn apply-to-map-values
176    "Applies function `f` to the values of the map `m`."
177    [f m]
178    (into {}
179          (map (fn [[k v]]
180                (cond
181                  (map? v) [k (apply-to-map-values f v)]
182                  :default [k (f v)]))
183               m)))
184  
185  (defn apply-to-map-keys
186    "Applies function `f` to the keys of the map `m`."
187    [f m]
188    (into {}
189          (map (fn [[k v]]
190                 (cond
191                   (map? v) [(f k) (apply-to-map-keys f v)]
192                   :default [(f k) v]))
193               m)))

208    "Replaces all of the `@`-signs (if followed by letters 1)
209    by a `%`-sign  because `:%kw` is a valid keyword but `:@kw` not
210    (or at least problematic).
211  
212    1) There are devices annotating channeles by `(@101:105)`.
