035  (defn short-string
036    "Short a `string` `s` to `n` or `45` chars. Returns `nil` is `s` is
037    not a `string`."
038    ([s]
039     (when (string? s) (short-string s 40)))
040    ([s n]
041     (when (string? s) (if (< (count s) n) s (str (subs s 0 n) "...")))))

043  (defn ensure-int
044    "Ensures `i` to be integer. Returns 0 as default.
045  
046    ```clojure
047    (u/ensure-int 100)
048    ;; 100
049    (u/ensure-int \"w\")
050    ;; 0
051    (u/ensure-int \"00\")
052    ;; 0
053    (u/ensure-int \"10\")
054    ;; 10
055    (u/ensure-int true)
056    ;; 0
057    ```
058    "
059    [i]
060    (if (integer? i)
061      i
062      (try (Integer/parseInt i) (catch Exception e 0))))

064  (defn pad-ok?
065    "Checks if the padding of `i` is ok. `\"*\"` serves pattern matching."
066    ([i]
067     (pad-ok? i (cfg/key-pad-length (cfg/config))))
068    ([i n]
069     (cond
070       (= i "*")         true 
071       (and
072        (string? i)
073        (= n (count i))) true
074       :else             false)))

076  (defn lp
077    "Left pad the given number if it is not a string. Default
078    is `(cfg/key-pad-length (cfg/config))`.
079  
080    Example:
081    ```clojure
082     (u/lp 2)
083    ;; \"002\"
084    (u/lp \"02\")
085    ;; \"002\"
086    (u/lp 2)
087    ;; \"002\"
088    (u/lp true)
089    ;; \"000\"
090    (u/lp \"003\")
091    ;; \"003\"
092    (u/lp \"000003\")
093    ;; \"003\"
094    ```
095    "
096    ([i]
097     (if (pad-ok? i) i (lp (ensure-int i) 3)))
098    ([i n]
099     (if (pad-ok? i) i (format (str "%0" n "d") (ensure-int i)))))

104  (defn extr-main-path
105    "Extracts the main path. Should work on `mpd-aaa-bbb` as well as on
106    `aaa-bbb`.
107  
108    ```clojure
109    (u/extr-main-path \"aa\")
110    ;; \"aa\"
111    (u/extr-main-path \"aa-bbb\")
112    ;; \"aa-bbb\"
113    (u/extr-main-path \"aa-bbb-lll\")
114    ;; \"aa-bbb-lll\"
115    ```
116    "
117    [s]
118    (let [p "mpd-"]
119      (if (string/starts-with? s p) (string/replace  s (re-pattern p) "") s)))

126    ;; \"mpd-aaa\"
127    (u/compl-main-path \"mpd-aaa\")

129    ```

136  (defn apply-to-map-values
137    "Applies function `f` to the values of the map `m`."
138    [f m]
139    (into {}
140          (map (fn [[k v]]
141                (cond
142                  (map? v) [k (apply-to-map-values f v)]
143                  :default [k (f v)]))
144               m)))
145  
146  (defn apply-to-map-keys
147    "Applies function `f` to the keys of the map `m`."
148    [f m]
149    (into {}
150          (map (fn [[k v]]
151                 (cond
152                   (map? v) [(f k) (apply-to-map-keys f v)]
153                   :default [(f k) v]))
154               m)))

168  (defn doc->safe-doc
169    "Replaces all of the `@`-signs (if followed by letters 1)
170    by a `%`-sign  because `:%kw` is a valid keyword but `:@kw` not
171    (or at least problematic).
172  
173    1) There are devices annotating channeles by `(@101:105)`.
174    This should remain as it is.
175    "
176    [doc]
177    (json->map (string/replace (map->json doc)  #"(@)([a-zA-Z])" "%$2")))
