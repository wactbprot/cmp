001  (ns cmp.server
002      ^{:author "wactbprot"
003        :doc "Provides a server for cmp info and ctrl. Starts up the configured mpds."}
004    (:require [compojure.route          :as route]
005              [com.brunobonacci.mulog   :as mu]
006              [cmp.config               :as config] 
007              [cmp.cli                  :as cli]
008              [cmp.api                  :as a]
009              [cmp.ui.index             :as ui]
010              [cmp.ui.listener          :as uil]
011              [cmp.ui.container         :as uic]
012              [cmp.ui.mp-meta           :as uim]
013              [cmp.ui.ws                :as ws]
014              [cmp.st-mem               :as st]
015              [compojure.core           :refer :all]
016              [compojure.handler        :as handler]
017              [org.httpkit.server       :refer [run-server]]
018              [ring.middleware.json     :as middleware]
019              [ring.util.response       :as res])
020      (:use   [clojure.repl]))
021  
022  (def conf (config/config))
023  
024  (defonce server (atom nil))
025  
026  (defroutes app-routes
027    (GET "/ui/listeners"                          [:as req] (uil/view conf req (a/listeners conf req)))
028    (GET "/ui"                                    [:as req] (uil/view conf req (a/listeners conf req)))
029    (GET "/ui/:mp/meta"                           [:as req] (uim/view conf req (a/mp-meta conf req)))
030    (GET "/ui/:mp"                                [:as req] (uim/view conf req (a/mp-meta conf req)))
031    (GET "/ui/:mp/container/ctrl"                 [:as req] (uic/view-ctrl  conf req (a/container-ctrl       conf req)))
032    (GET "/ui/:mp/container/state"                [:as req] (uic/view-state conf req (a/container-state      conf req)))
033    (GET "/ui/:mp/container/definition"           [:as req] (uic/view       conf req (a/container-definition conf req)))
034    (GET "/ui/:mp/container/ctrl/:idx"            [:as req] (uic/view-ctrl  conf req (a/container-ctrl       conf req)))
035    (GET "/ui/:mp/container/state/:idx"           [:as req] (uic/view-state conf req (a/container-state      conf req)))
036    (GET "/ui/:mp/container/definition/:idx"      [:as req] (uic/view       conf req (a/container-definition conf req)))
037    (GET "/ui/:mp/container/state/:idx/:seq"      [:as req] (uic/view-state conf req (a/container-state      conf req)))
038    (GET "/ui/:mp/container/definition/:idx/:seq" [:as req] (uic/view       conf req (a/container-definition conf req)))
039    (POST "/:mp/container"                        [:as req] (res/response (a/set-val! conf req)))
040    (GET "/ws"                                    [:as req] (ws/main        conf req))
041    (route/resources "/")
042    (route/not-found (res/response {:error "not found"})))
043  
044  (def app
045    (-> (handler/site app-routes)
046        (middleware/wrap-json-body {:keywords? true})
047        middleware/wrap-json-response))
048  
049  (defn stop []
050    (mu/log ::stop :message "stop ui web socket listener")
051    (ws/stop! conf)
052    (run! (fn [mp-id]
053            (mu/log ::stop :message "stop mpd" :mp-id mp-id)
054            (cli/m-stop conf mp-id))
055          (:build-on-start conf))
056    (when @server (@server :timeout 100)
057          (mu/log ::stop :message "stop server")
058          (reset! server nil)
059          (mu/log ::stop :message "stop logger")
060          (cli/stop-log! conf)))
061  
062  (defn start []
063    (cli/start-log! conf)
064    (mu/log ::start :message "start ui web socket listener")
065    (ws/start! conf)
066    (mu/log ::start :message "refresh tasks")
067    (cli/t-refresh conf)
068    (run! (fn [mp-id]
069            (mu/log ::start :message "build mpd" :mp-id mp-id)
070            (cli/m-build-ltm conf mp-id))
071          (:build-on-start conf))
072    (mu/log ::start :message "start server")
073    (reset! server (run-server #'app (:api conf))))
074  
075  (defn restart [] (stop) (Thread/sleep 1000) (start))
076  
077  (defn -main [& args]
078    (mu/log ::main :message "call -main")
079    (start))
