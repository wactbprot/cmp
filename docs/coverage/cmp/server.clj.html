001  (ns cmp.server
002      ^{:author "wactbprot"
003        :doc "Provides a REST api for cmp info and ctrl."}
004    (:require [compojure.route          :as route]
005              [com.brunobonacci.mulog   :as mu]
006              [cmp.config               :as c]
007              [cmp.api                  :as a]
008              [cmp.ui.core              :as ui]
009              [cmp.ui.listener          :as uil]
010              [cmp.ui.container         :as uic]
011              [cmp.ui.mp-meta           :as uim]
012              [cmp.ui.ws                :as ws]
013              [cmp.st-mem               :as st] 
014              [compojure.core           :refer :all]
015              [compojure.handler        :as handler]
016              [org.httpkit.server       :refer [run-server]]
017              [ring.middleware.json     :as middleware]
018              [ring.util.response       :as res]
019              ))
020  
021  (def conf (c/config))
022  
023  (defonce server (atom nil))
024  
025  (defonce logger (atom nil))
026  
027  (defroutes app-routes
028    (GET "/config"                      []           (res/response conf))
029    (GET "/listeners"                   [:as req]    (res/response (a/listeners conf req)))
030    (GET "/tasks"                       [:as req]    (res/response (a/tasks     conf req)))
031    
032    (GET "/:mp/meta"                    [mp :as req] (res/response (a/mp-meta   conf req mp)))
033    (GET "/:mp/container/title"         [mp :as req] (res/response (a/container-title conf req mp)))
034  
035    (GET "/ui/listeners"                   [:as req] (uil/view conf (a/listeners conf req)))
036  
037    (GET "/ui/:mp/meta"                 [mp :as req] (uim/view conf
038                                                               (a/mp-meta conf req mp)))
039    (GET "/ui/:mp/container/title"      [mp :as req] (uic/view conf
040                                                               (a/container-title conf req mp)))
041    (GET "/ui/:mp/container/descr"      [mp :as req] (uic/view conf
042                                                               (a/container-descr conf req mp)))
043    (GET "/ui/:mp/container/ctrl"       [mp :as req] (uic/view conf
044                                                               (a/container-ctrl conf req mp)))
045    (GET "/ui/:mp/container/state"      [mp :as req] (uic/view conf
046                                                               (a/container-state conf req mp)))
047    (GET "/ui/:mp/container/definition" [mp :as req] (uic/view conf
048                                                               (a/container-definition conf req mp)))
049  
050    (GET "/ws"                     [:as req] (ws/main  conf req))
051    
052    (route/resources "/")
053    (route/not-found (res/response {:error "not found"})))
054  
055  (def app
056    (-> (handler/site app-routes)
057        (middleware/wrap-json-body {:keywords? true})
058        middleware/wrap-json-response))
059  
060  (defn start-ws!
061    [conf]
062    (st/register! "*" "*" "*" "*"  (fn [msg]
063                                     (when-let [k (st/msg->key msg)]
064                                       (ws/send-to-ws-clients conf {:key (ui/make-selectable k) :value (st/key->val k)}))) "c"))
065    
066  (defn stop-ws! [conf] (st/de-register! "*" "*" "*" "*"  "c"))
067  
068  (defn init-log!
069    [{conf :mulog }]
070    (mu/set-global-context! {:app-name "cmp"})
071    (mu/start-publisher! conf))
072  
073  (defn stop []
074    (stop-ws! conf)
075    (when @server (@server :timeout 100)
076          (@logger)
077          (reset! logger nil)
078          (reset! server nil)))
079  
080  (defn start []
081    (start-ws! conf)
082    (reset! logger (init-log! conf))
083    (mu/log ::start :message "start cmp rest api")
084    (reset! server (run-server #'app (:api conf))))
085  
086  
087  (defn -main [& args]
088    (mu/log ::start :message "call -main")
089    (start))
