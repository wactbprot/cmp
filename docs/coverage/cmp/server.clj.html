001  (ns cmp.server
002      ^{:author "wactbprot"
003        :doc "Provides a REST api for cmp info and ctrl."}
004    (:require [compojure.route          :as route]
005              [com.brunobonacci.mulog   :as mu]
006              [cmp.config               :as c]
007              [cmp.api                  :as a]
008              [compojure.core           :refer :all]
009              [compojure.handler        :as handler]
010              [org.httpkit.server       :refer [run-server]]
011              [ring.middleware.json     :as middleware]
012              [ring.util.response       :as res]
013              ))
014  
015  (def conf (c/config))
016  
017  (defonce server (atom nil))
018  
019  (defonce logger (atom nil))
020  
021  (defroutes app-routes
022    (GET "/config"    []        (res/response conf))
023    (GET "/listeners" [:as req] (res/response (a/listeners conf req)))
024    (route/not-found            (res/response {:error "not found"})))
025  
026  (def app
027    (-> (handler/site app-routes)
028        (middleware/wrap-json-body {:keywords? true})
029        middleware/wrap-json-response))
030  
031  (defn init-log!
032    [{conf :mulog }]
033    (mu/set-global-context! {:app-name "cmp"})
034    (mu/start-publisher! conf))
035  
036  (defn stop []
037    (when @server (@server :timeout 100)
038          (@logger)
039          (reset! logger nil)
040          (reset! server nil)))
041  
042  (defn start []
043    (reset! logger (init-log! conf))
044    (mu/log ::start :message "start cmp rest api")
045    (reset! server (run-server app (:api conf))))
046  
047  
048  (defn -main [& args]
049    (mu/log ::start :message "call -main")
050    (start))
