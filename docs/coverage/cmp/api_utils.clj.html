010  (defn key-value-map
011    "Generates a map consisting of `k`ey and `v`alue. The result is
012    merged with `m`."
013    ([k]
014     (key-value-map k nil))
015    ([k m]
016     (merge (assoc (stu/key->info-map k) :value (st/key->val k) :key k) m)))
017  
018  (defn req->mp-id [req] (get-in req [:route-params :mp] "*"))
019  
020  (defn req->seq-idx [req] (get-in req [:route-params :seq] "*"))
021  
022  (defn req->no-idx
023    "Returns the `no-idx` (container number). Uses
024    `run-mp/title->no-idx` if the idx rout-param is not `[0-9]*`"
025    [req]
026    (let [s (get-in req [:route-params :idx])]
027      (cond
028        (nil? s) "*"
029        (re-matches #"[0-9]*" s)  s
030        :title  (codec/url-encode (ctrl-mp/title->no-idx (req->mp-id req) s)))))
