008  (defn key-value-map
009    "Generates a map consisting of `k`ey and `v`alue. The result is
010    merged with `m`."
011    ([k]
012     (key-value-map k nil))
013    ([k m]
014     (merge (assoc (ku/key->info-map k) :value (st/key->val k) :key k) m)))
