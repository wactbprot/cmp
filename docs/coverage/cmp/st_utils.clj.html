011  (def sep
012    "Short-term-database (st) path seperator.
013    Must not be a regex operator (like `.` or `|`)"
014    "@")
015  
016  (def re-sep
017    "The regex version of the seperator."
018    (re-pattern sep))
019  
020  (defn vec->key
021    "Joins the vec to a key."
022    [v]
023    (string/join sep v))
024  
025  (defn key-at-level
026    "Returns the value of the key `k` at the level `l`."
027    [k l]
028    {:pre [(string? k)
029           (int? l)] }
030    (nth (string/split k re-sep) l nil))
031  
032  (defn replace-key-at-level
033    "Generates a new key by replacing an old key `k` at the given position
034    `l` with the given string `r`.
035  
036    REVIEW The key levels should have a name or keyword.  Passing
037    integers (`l`) is unimaginative.
038    "
039    [l k r]
040    {:pre [(string? k)
041           (int? l)
042           (string? r)]}
043    (let [v (string/split k re-sep)]
044      (when (< l (count v)) (vec->key (assoc v l r)))))

134  (defn key->info-map
135    "Builds a `info-map` out of the key structure.
136    
137    Example:
138    ```clojure
139    (key->info-map \"\") 
140    ;;{:mp-id nil,
141    ;; :struct nil,
142    ;; :no-idx nil,
143    ;; :func nil,
144    ;; :seq-idx nil,
145    ;; :par-idx nil}
146    ```"
147    [k]
148    (when (string? k)
149      {:mp-id   (key->mp-id   k)
150       :struct  (key->struct  k)
151       :func    (key->func    k)
152       :no-idx  (key->no-idx  k)
153       :seq-idx (key->seq-idx k)
154       :par-idx (key->par-idx k)}))

172  (defn seq-idx->all-par
173    "Returns a vector of [[info-maps]] with all `par` steps for a given
174    `i`.
175  
176    Example:
177    ```clojure
178      (seq-idx->all-par [{:seq-idx 0 :par-idx 0 }
179                         {:seq-idx 1 :par-idx 0 }
180                         {:seq-idx 1 :par-idx 1 }] 1)
181    ;; =>
182    ;; [{:seq-idx 1 :par-idx 0 :state :ready}
183    ;;  {:seq-idx 1 :par-idx 1 :state :ready}]
184    ```"
185    [v i]
186    (filterv (fn [m] (= (u/ensure-int i)
187                        (u/ensure-int (:seq-idx m)))) v))
