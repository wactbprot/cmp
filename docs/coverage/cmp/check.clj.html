001  (ns cmp.check
002    (:require [cmp.utils :as u]
003              [cmp.st-mem :as st]
004              [cmp.task :as tsk]
005              [taoensso.timbre :as log]))
006  
007  (defn struct-tasks
008    "Checks the tasks of the structure (`container` or `definitions`):
009  
010    * the `proto-task` is taken from the definition section
011    * the `meta-task` is gathered from different sources
012    * the `state` at the path is set to `checking`
013    * the `meta-task` is checked against the 
014    * the `state` is set to `ready`"   
015    [p]
016    (run!
017     (fn [k]
018       (let [state-key   (u/replace-key-at-level 3 k "state")
019             proto-task  (st/key->val k)
020             meta-task   (tsk/gen-meta-task proto-task)]
021         (st/set-val! state-key "checking")
022         (assert (tsk/meta-task? meta-task))
023         (st/set-val! state-key "ready")))
024     (st/key->keys p)))
