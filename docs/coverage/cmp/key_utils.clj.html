013  (defn key->mp-id
014    "Returns the name of the key space for the given key.
015  
016    May be:
017    * tasks
018    * <mp-id>
019  
020    "
021    [k]
022    (when (and
023           (string? k)
024           (not (empty? k)))
025      (nth (string/split k u/re-sep) 0 nil)))

028  (defn key->struct
029    "Returns the name of the `struct`ure for the given key.
030    The structure is the name of the key at the second
031    position. Possible values are:
032    
033    * `<taskname>`
034    * `definitions`
035    * `container`
036    * `exchange`
037    * `id`
038    * `meta`
039    "
040    [k]
041    (when (string? k)
042      (nth (string/split k u/re-sep) 1 nil)))

093  (defn key->info-map
094    "Builds a `info-map` out of the key structure.
095    
096    Example:
097    ```clojure
098    (key->info-map \"\") 
099    ;;{:mp-id nil,
100    ;; :struct nil,
101    ;; :no-idx nil,
102    ;; :func nil,
103    ;; :seq-idx nil,
104    ;; :par-idx nil}
105    ```"
106    [k]
107    (when (string? k)
108      {:mp-id   (key->mp-id   k)
109       :struct  (key->struct  k)
110       :func    (key->func    k)
111       :no-idx  (key->no-idx  k)
112       :seq-idx (key->seq-idx k)
113       :par-idx (key->par-idx k)}))
114  
115  (defn info-map->definition-key
116    "Converts a `state-map` into the related `definition` key."
117    [m]
118    (when (map? m)
119      (u/vec->key [(:mp-id m)
120                   (:struct m)
121                   (:no-idx m)
122                   "definition"
123                   (:seq-idx m)
124                   (:par-idx m)])))
125   
126  (defn seq-idx->all-par
127    "Returns a vector of [[info-maps]] with all `par` steps for a given
128    `i`.
129  
130    Example:
131    ```clojure
132      (seq-idx->all-par [{:seq-idx 0 :par-idx 0 }
133                         {:seq-idx 1 :par-idx 0 }
134                         {:seq-idx 1 :par-idx 1 }] 1)
135    ;; =>
136    ;; [{:seq-idx 1 :par-idx 0 :state :ready}
137    ;;  {:seq-idx 1 :par-idx 1 :state :ready}]
138  
139    REVIEW: does this function really belong to ku ns?
140    ```"
141    [v i]
142    (filterv (fn [m] (= (u/ensure-int i)
143                        (u/ensure-int (:seq-idx m)))) v))

146    "Turns the given `state-key` into a `response-key`.
147  
148      Example:
149    ```clojure
150    (key->response-key \"ref@container@0@state@0@0\")
151    ;; devs@container@0@response@0@0
152    ```
153    "
154    [k]
155    (u/replace-key-at-level 3 k "response"))

186  (defn key->ctrl-key
187    "Returns the `ctrl`-key for a given key `k`.
188    In other words: ensures k to be a `ctrl-key`.
189  
190    ```clojure
191    (key->ctrl-key \"wait@container@0@state@0@0\")
192    ;; \"wait@container@0@ctrl\"
193    (key->ctrl-key
194      (key->ctrl-key
195        (key->ctrl-key \"wait@container@0@state@0@0\")))
196      ;; \"wait@container@0@ctrl\"
197    ```" 
198    [k]
199    (u/vec->key [(key->mp-id k)
200                 (key->struct k)
201                 (key->no-idx k)
202                 "ctrl"]))
