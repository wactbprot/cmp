(compl-main-path s)

;; "mpd-aaa"
(u/compl-main-path "mpd-aaa")

extr-main-path

(extr-main-path s)

Extracts the main path.

Should work on mpd-aaa-bbb as well as on aaa-bbb.

(u/extr-main-path "aa")
;; "aa"
cmp.core> (u/extr-main-path "aa-bbb")
;; "aa-bbb"
cmp.core> (u/extr-main-path "aa-bbb-lll")
